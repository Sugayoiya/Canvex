    ```typescript
    const queryParams = {
      q: debouncedSearch || undefined,
      limit: pagination.pageSize,
      offset: pagination.pageIndex * pagination.pageSize,
    };
    const { data, isLoading, isError, refetch } = useQuery({
      queryKey: ["admin", "teams", queryParams],
      queryFn: () => adminApi.listTeams(queryParams).then(r => r.data as AdminTeamListResponse),
    });
    ```

    3. `owner_name` — header "OWNER", width 180px, cell: show owner nickname or "—" in `--cv4-text-muted` when null, Manrope 12px/400 `--cv4-text-secondary`, enableSorting false
    4. `created_at` — header "CREATED", width 160px, cell: formatted date string, Manrope 12px/400 `--cv4-text-secondary`, enableSorting false

    No `actions` column. No sorting state (per D-04 read-only, no sort on teams table).

    **TanStack Table instance:**
    ```typescript
    const table = useReactTable({
      data: data?.items ?? [],
      columns,
      pageCount: Math.ceil((data?.total ?? 0) / pagination.pageSize),
    ```

    ```
    <div style={{ display: "flex", flexDirection: "column", gap: 16 }}>
      <!-- PageHeader: "Teams" title + "View all teams and their members" subtitle -->

        searchValue={searchValue}
        onSearchChange={setSearchValue}
        searchPlaceholder="Search by team name..."
      />
      <!-- No filters prop = no filter dropdowns, only search -->
      <AdminDataTable table={table} isLoading={isLoading} isError={isError} onRetry={refetch}
        emptyIcon={debouncedSearch ? <Search size={48}/> : <Group size={48}/>}

      )}
    </div>
    ```
