"""
Phase 02 Integration Acceptance Test
Validates: 5 node types + skill execution + AICallLog + billing cost
"""
import pytest
from decimal import Decimal

@pytest.mark.asyncio
async def test_all_skills_registered():
    """All Phase 02 skill categories are registered."""
    from app.skills.register_all import register_all_skills
    from app.skills.registry import skill_registry
    register_all_skills()
    names = skill_registry.list_names()
    # TEXT (4) + EXTRACT (2) + SCRIPT (2) + STORYBOARD (2) + VISUAL (3) + existing (canvas_ops, asset)
    assert len(names) >= 15, f"Expected >= 15 skills, got {len(names)}: {names}"
    for expected in ["text.llm_generate", "text.refine", "extract.characters", "extract.scenes",
                     "script.split_clips", "storyboard.plan", "visual.character_prompt", "visual.generate_image"]:
        assert expected in names, f"Skill '{expected}' not found"

@pytest.mark.asyncio
async def test_text_skill_execution(mock_provider_manager, skill_context):
    """TEXT skill invokes mock provider and returns SkillResult."""
    from app.skills.register_all import register_all_skills
    from app.skills.registry import skill_registry
    register_all_skills()
    result = await skill_registry.invoke("text.llm_generate", {"prompt": "测试输入"}, skill_context)
    assert result.status == "completed"
    assert "text" in result.data
    assert len(result.data["text"]) > 0

@pytest.mark.asyncio
async def test_extract_skill_returns_json(mock_provider_manager, skill_context):
    """EXTRACT skill returns parsed JSON data."""
    from app.skills.register_all import register_all_skills
    from app.skills.registry import skill_registry
    register_all_skills()
    result = await skill_registry.invoke("extract.characters", {"text": "张三是一个年轻的程序员"}, skill_context)
    assert result.status == "completed"
    assert "characters" in result.data
    assert isinstance(result.data["characters"], list)

@pytest.mark.asyncio
async def test_sync_visual_skill():
    """Sync visual skill executes without provider call."""
    from app.skills.register_all import register_all_skills
    from app.skills.registry import skill_registry
    from app.skills.context import SkillContext
    register_all_skills()
    ctx = SkillContext(trace_id="test", user_id="test")
    result = await skill_registry.invoke(
        "visual.character_prompt",
        {"character": {"name": "Alice", "description": "a warrior", "gender": "female", "age": "25"}},
        ctx,
    )
    assert result.status == "completed"
    assert "prompt" in result.data
    assert "Alice" in result.data["prompt"]

@pytest.mark.asyncio
async def test_canvas_model_creation(async_session):
    """Canvas + nodes can be created in DB."""
    from app.models.canvas import Canvas, CanvasNode
    import uuid
    canvas = Canvas(id=str(uuid.uuid4()), name="Test Canvas", project_id="proj-001", created_by="user-001")
    async_session.add(canvas)
    await async_session.flush()

    for node_type in ["text-input", "llm-generate", "extract", "image-gen", "output"]:
        node = CanvasNode(id=str(uuid.uuid4()), canvas_id=canvas.id, node_type=node_type)
        async_session.add(node)
    await async_session.commit()

    from sqlalchemy import select, func
    count = (await async_session.execute(
        select(func.count()).select_from(CanvasNode).where(CanvasNode.canvas_id == canvas.id)
    )).scalar()
    assert count == 5, f"Expected 5 nodes, got {count}"

@pytest.mark.asyncio
async def test_billing_cost_calculation():
    """calculate_cost returns correct Decimal for per_token pricing."""
    from app.services.billing.pricing_service import calculate_cost
    class MockPricing:
        pricing_model = "per_token"
        input_price_per_1k = Decimal("0.0001")
        output_price_per_1k = Decimal("0.0004")
        price_per_image = None; price_per_second = None; price_per_request = None
    cost = calculate_cost(MockPricing(), {"input_tokens": 1000, "output_tokens": 500})
    expected = Decimal("0.0001") + Decimal("0.0002")  # 1k * 0.0001 + 0.5k * 0.0004
    assert cost == expected, f"Expected {expected}, got {cost}"

@pytest.mark.asyncio
async def test_billing_image_cost():
    """calculate_cost handles per_image pricing."""
    from app.services.billing.pricing_service import calculate_cost
    class MockPricing:
        pricing_model = "per_image"
        price_per_image = Decimal("0.04")
        input_price_per_1k = None; output_price_per_1k = None
        price_per_second = None; price_per_request = None
    cost = calculate_cost(MockPricing(), {"image_count": 2})
    assert cost == Decimal("0.08")
