Handler `handle_generate_image`:
```python
async def handle_generate_image(params: dict, ctx: SkillContext) -> SkillResult:
    prompt = params.get("prompt", "")
    if not prompt:
        return SkillResult.failed("prompt 不能为空")

    size = params.get("size", "1024x1024")
    quality = params.get("quality", "standard")

    from app.services.ai.ai_call_logger import set_ai_call_context, log_ai_call
    set_ai_call_context(trace_id=ctx.trace_id, user_id=ctx.user_id,
                        team_id=ctx.team_id, project_id=ctx.project_id)

    from app.core.config import settings
    import time

    if not settings.OPENAI_API_KEY:
        return SkillResult.failed("OPENAI_API_KEY not configured (required for image generation)")

    from openai import AsyncOpenAI
    client = AsyncOpenAI(api_key=settings.OPENAI_API_KEY)

    start = time.monotonic()
    try:
        response = await client.images.generate(
            model="dall-e-3", prompt=prompt, size=size, quality=quality, n=1,
        )
        duration_ms = int((time.monotonic() - start) * 1000)
        image_url = response.data[0].url
        revised_prompt = response.data[0].revised_prompt or prompt

        await log_ai_call(
            provider="openai", model="dall-e-3", model_type="image",
            duration_ms=duration_ms, status="success",
            input_summary=prompt[:200],
            trace_id=ctx.trace_id, user_id=ctx.user_id,
            team_id=ctx.team_id, project_id=ctx.project_id,
        )

        return SkillResult(
            status="completed",
            data={"image_url": image_url, "revised_prompt": revised_prompt},
            artifacts=[{"type": "image", "url": image_url}],
            message="图像生成完成",
        )
    except Exception as e:
        duration_ms = int((time.monotonic() - start) * 1000)
        await log_ai_call(
            provider="openai", model="dall-e-3", model_type="image",
            duration_ms=duration_ms, status="failed", error_message=str(e),
            trace_id=ctx.trace_id, user_id=ctx.user_id,
        )
        return SkillResult.failed(f"图像生成失败: {str(e)}")
```
