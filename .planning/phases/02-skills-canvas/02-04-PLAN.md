Handler `handle_split_clips`:
```python
async def handle_split_clips(params, ctx):
    text = params.get("text", "")
    if not text:
        return SkillResult.failed("text 不能为空")
    max_clips = params.get("max_clips", 20)
    from app.skills.llm_utils import call_llm, parse_llm_json
    result_text, usage = await call_llm(
        ctx,
        f"你是专业的影视分镜师。请将以下文本切分为{max_clips}个以内的独立叙事片段。"
        '返回 JSON 数组: [{"index": 1, "content": "片段原文", "summary": "一句话概要"}]。只返回 JSON。',
        text,
    )
    clips = parse_llm_json(result_text, list)
    if clips is None:
        clips = [{"index": 1, "content": text[:500], "summary": "解析失败"}]
    return SkillResult(status="completed", data={"clips": clips},
                       message=f"切分为 {len(clips)} 个片段")
```
