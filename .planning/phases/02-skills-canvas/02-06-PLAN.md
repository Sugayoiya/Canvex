```python
from decimal import Decimal
from sqlalchemy import select, and_
from datetime import datetime

async def get_active_pricing(provider: str, model: str) -> "ModelPricing | None":
    from app.core.database import AsyncSessionLocal
    from app.models.model_pricing import ModelPricing
    now = datetime.utcnow()
    async with AsyncSessionLocal() as session:
        stmt = select(ModelPricing).where(
            and_(
                ModelPricing.provider == provider,
                ModelPricing.model == model,
                ModelPricing.is_active == True,
                ModelPricing.effective_from <= now,
                (ModelPricing.effective_to == None) | (ModelPricing.effective_to >= now),
            )
        ).order_by(ModelPricing.effective_from.desc()).limit(1)
        result = await session.execute(stmt)
        return result.scalar_one_or_none()

def calculate_cost(pricing, usage: dict) -> Decimal:
    """Calculate cost from a ModelPricing row and usage dict.
    usage keys: input_tokens, output_tokens, image_count, duration_seconds.
    """
    if pricing is None:
        return Decimal("0")

    if pricing.pricing_model == "per_token":
        input_cost = Decimal(str(usage.get("input_tokens", 0))) / Decimal("1000") * (pricing.input_price_per_1k or Decimal("0"))
        output_cost = Decimal(str(usage.get("output_tokens", 0))) / Decimal("1000") * (pricing.output_price_per_1k or Decimal("0"))
        return input_cost + output_cost
    elif pricing.pricing_model == "per_image":
        return (pricing.price_per_image or Decimal("0")) * Decimal(str(usage.get("image_count", 1)))
    elif pricing.pricing_model == "per_second":
        return (pricing.price_per_second or Decimal("0")) * Decimal(str(usage.get("duration_seconds", 0)))
    elif pricing.pricing_model == "per_request":
        return pricing.price_per_request or Decimal("0")
    return Decimal("0")

async def calculate_cost_for_call(provider: str, model: str, usage: dict) -> Decimal:
    pricing = await get_active_pricing(provider, model)
    return calculate_cost(pricing, usage)
```

async def seed_default_pricing():
    from app.core.database import AsyncSessionLocal
    from app.models.model_pricing import ModelPricing
    from sqlalchemy import select, func
    async with AsyncSessionLocal() as session:
        count = (await session.execute(select(func.count()).select_from(ModelPricing))).scalar() or 0
        if count == 0:
            for data in SEED_PRICING:
                session.add(ModelPricing(**data))
            await session.commit()
