    ```typescript
    "use client";
    import { useState, useRef, useCallback, useEffect } from "react";
    import { skillsApi } from "@/lib/api";
    import { useCanvasStore } from "@/stores/canvas-store";

    interface ExecutionState {
      status: "idle" | "running" | "completed" | "failed";
      data: any;
      message: string;
      progress: number;
    }

    const INITIAL_POLL_INTERVAL = 3000;   // 3s
    const MAX_POLL_INTERVAL = 15000;      // 15s
    const MAX_POLL_ATTEMPTS = 60;
    const BACKOFF_MULTIPLIER = 1.5;

    export function useNodeExecution(nodeId: string) {
      const [state, setState] = useState<ExecutionState>({
        status: "idle", data: null, message: "", progress: 0,
      });
      const pollTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
      const pollCountRef = useRef(0);
      const pollIntervalRef = useRef(INITIAL_POLL_INTERVAL);
      const isMountedRef = useRef(true);

      const { canvasId, projectId } = useCanvasStore();

      // Cleanup on unmount
      useEffect(() => {
        isMountedRef.current = true;
        return () => {
          isMountedRef.current = false;
          if (pollTimerRef.current) {
            clearTimeout(pollTimerRef.current);
            pollTimerRef.current = null;
          }
        };
      }, []);

      const pollResult = useCallback(async (taskId: string) => {
        if (!isMountedRef.current) return;
        if (pollCountRef.current >= MAX_POLL_ATTEMPTS) {
          setState(prev => ({ ...prev, status: "failed", message: "轮询超时，请重试" }));
          return;
        }

        try {
          const res = await skillsApi.poll(taskId);
          if (!isMountedRef.current) return;
          const { status, data, message, progress } = res.data;

          if (status === "completed") {
            setState({ status: "completed", data, message, progress: 1 });
            return;
          }
          if (status === "failed") {
            setState({ status: "failed", data, message, progress: 0 });
            return;
          }

          // Still running — schedule next poll with backoff
          setState(prev => ({ ...prev, progress: progress || prev.progress, message }));
          pollCountRef.current += 1;
          pollIntervalRef.current = Math.min(
            pollIntervalRef.current * BACKOFF_MULTIPLIER,
            MAX_POLL_INTERVAL,
          );
          pollTimerRef.current = setTimeout(() => pollResult(taskId), pollIntervalRef.current);
        } catch (err: any) {
          if (!isMountedRef.current) return;
          setState(prev => ({ ...prev, status: "failed", message: `轮询出错: ${err?.message || "未知错误"}` }));
        }
      }, []);

      const execute = useCallback(async (skillName: string, params: Record<string, any> = {}) => {
        setState({ status: "running", data: null, message: "正在执行...", progress: 0 });
        pollCountRef.current = 0;
        pollIntervalRef.current = INITIAL_POLL_INTERVAL;

        try {
          const res = await skillsApi.invoke({
            skill_name: skillName,
            params,
            project_id: projectId || undefined,
            canvas_id: canvasId || undefined,
            node_id: nodeId,
          });

          if (!isMountedRef.current) return;
          const { status, data, message, task_id } = res.data;

          if (status === "completed") {
            setState({ status: "completed", data, message, progress: 1 });
          } else if (status === "failed") {
            setState({ status: "failed", data, message, progress: 0 });
          } else if (status === "running" && task_id) {
            // Start polling
            pollTimerRef.current = setTimeout(() => pollResult(task_id), INITIAL_POLL_INTERVAL);
          }
        } catch (err: any) {
          if (!isMountedRef.current) return;
          setState({ status: "failed", data: null, message: `调用失败: ${err?.message || "未知错误"}`, progress: 0 });
        }
      }, [nodeId, canvasId, projectId, pollResult]);

      const reset = useCallback(() => {
        if (pollTimerRef.current) {
          clearTimeout(pollTimerRef.current);
          pollTimerRef.current = null;
        }
        setState({ status: "idle", data: null, message: "", progress: 0 });
      }, []);

      return { ...state, execute, reset };
    }
    ```

    Key implementation details per CONTEXT.md decisions:

    - Max interval: 15s (MAX_POLL_INTERVAL = 15000)
    - Max attempts: 60 (MAX_POLL_ATTEMPTS = 60)
    - Cleanup: isMountedRef + clearTimeout in useEffect cleanup
