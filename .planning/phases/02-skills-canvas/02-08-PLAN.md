**web/src/hooks/use-node-execution.ts:**

Addresses review concern #7: exponential backoff, max attempts, unmount cleanup.

```typescript
import { useState, useCallback, useRef, useEffect } from "react";
import { skillsApi } from "@/lib/api";

interface ExecutionState {
  status: "idle" | "running" | "completed" | "failed";
  result: Record<string, unknown> | null;
  error: string | null;
  progress: number;
}

const INITIAL_POLL_MS = 3000;
const BACKOFF_FACTOR = 1.5;
const MAX_POLL_MS = 15000;
const MAX_ATTEMPTS = 30;

export function useNodeExecution(canvasId: string, nodeId: string) {
  const [state, setState] = useState<ExecutionState>({
    status: "idle", result: null, error: null, progress: 0,
  });
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const attemptRef = useRef(0);
  const abortedRef = useRef(false);

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      abortedRef.current = true;
      if (timerRef.current) clearTimeout(timerRef.current);
    };
  }, []);

  const pollWithBackoff = useCallback((taskId: string, currentDelay: number, resolve: (v: any) => void) => {
    if (abortedRef.current) return;
    attemptRef.current += 1;
    if (attemptRef.current > MAX_ATTEMPTS) {
      setState({ status: "failed", result: null, error: "Polling timeout: max attempts exceeded", progress: 0 });
      resolve(null);
      return;
    }

    timerRef.current = setTimeout(async () => {
      if (abortedRef.current) return;
      try {
        const poll = await skillsApi.poll(taskId);
        if (abortedRef.current) return;
        if (poll.data.status === "completed") {
          setState({ status: "completed", result: poll.data.data, error: null, progress: 1 });
          resolve(poll.data.data);
        } else if (poll.data.status === "failed") {
          setState({ status: "failed", result: null, error: poll.data.message, progress: 0 });
          resolve(null);
        } else {
          setState(prev => ({ ...prev, progress: poll.data.progress || prev.progress }));
          const nextDelay = Math.min(currentDelay * BACKOFF_FACTOR, MAX_POLL_MS);
          pollWithBackoff(taskId, nextDelay, resolve);
        }
      } catch (err) {
        if (abortedRef.current) return;
        setState({ status: "failed", result: null, error: String(err), progress: 0 });
        resolve(null);
      }
    }, currentDelay);
  }, []);

  const execute = useCallback(async (skillName: string, params: Record<string, unknown>) => {
    abortedRef.current = false;
    attemptRef.current = 0;
    setState({ status: "running", result: null, error: null, progress: 0 });

    try {
      const res = await skillsApi.invoke({
        skill_name: skillName, params, canvas_id: canvasId, node_id: nodeId,
      });
      const data = res.data;
      if (data.status === "completed") {
        setState({ status: "completed", result: data.data, error: null, progress: 1 });
        return data.data;
      }
      if (data.status === "failed") {
        setState({ status: "failed", result: null, error: data.message, progress: 0 });
        return null;
      }
      if (data.task_id) {
        return new Promise(resolve => pollWithBackoff(data.task_id, INITIAL_POLL_MS, resolve));
      }
    } catch (err) {
      setState({ status: "failed", result: null, error: String(err), progress: 0 });
      return null;
    }
  }, [canvasId, nodeId, pollWithBackoff]);

  const reset = useCallback(() => {
    abortedRef.current = true;
    if (timerRef.current) clearTimeout(timerRef.current);
    setState({ status: "idle", result: null, error: null, progress: 0 });
  }, []);

  return { ...state, execute, reset };
}
```
