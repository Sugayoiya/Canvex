```tsx
"use client";
import { useCallback, useEffect } from "react";
import {
  ReactFlow, useNodesState, useEdgesState, addEdge,
  Background, Controls, MiniMap,
  type Connection, type Edge,
} from "@xyflow/react";
import "@xyflow/react/dist/style.css";
import { nodeTypes } from "./nodes";
import { isValidConnection } from "@/lib/connection-rules";
import { CanvasToolbar } from "./canvas-toolbar";
import { canvasApi } from "@/lib/api";
import { useCanvasStore } from "@/stores/canvas-store";

interface CanvasWorkspaceProps {
  canvasId: string;
  initialNodes?: any[];
  initialEdges?: any[];
}

export function CanvasWorkspace({ canvasId, initialNodes = [], initialEdges = [] }: CanvasWorkspaceProps) {
  // Convert backend nodes to xyflow format
  const xyNodes = initialNodes.map(n => ({
    id: n.id,
    type: n.node_type,
    position: { x: n.position_x, y: n.position_y },
    data: { ...n.data, label: n.label, status: n.status, canvasId },
  }));
  const xyEdges = initialEdges.map(e => ({
    id: e.id,
    source: e.source_node_id,
    target: e.target_node_id,
    sourceHandle: e.source_handle,
    targetHandle: e.target_handle,
  }));

  const [nodes, setNodes, onNodesChange] = useNodesState(xyNodes);
  const [edges, setEdges, onEdgesChange] = useEdgesState(xyEdges);

  const onConnect = useCallback((connection: Connection) => {
    setEdges((eds) => addEdge(connection, eds));
    // Persist edge to backend
    if (connection.source && connection.target) {
      canvasApi.addEdge(canvasId, {
        source_node_id: connection.source,
        target_node_id: connection.target,
        source_handle: connection.sourceHandle ?? undefined,
        target_handle: connection.targetHandle ?? undefined,
      });
    }
  }, [canvasId, setEdges]);

  const onAddNode = useCallback(async (nodeType: string) => {
    const res = await canvasApi.addNode(canvasId, {
      node_type: nodeType,
      label: nodeType,
      position_x: 100 + Math.random() * 400,
      position_y: 100 + Math.random() * 300,
    });
    const n = res.data;
    setNodes((nds) => [...nds, {
      id: n.id,
      type: n.node_type,
      position: { x: n.position_x, y: n.position_y },
      data: { ...n.data, label: n.label, status: n.status, canvasId },
    }]);
  }, [canvasId, setNodes]);

  return (
    <div className="h-full w-full flex flex-col">
      <CanvasToolbar onAddNode={onAddNode} />
      <div className="flex-1">
        <ReactFlow
          nodes={nodes}
          edges={edges}
          nodeTypes={nodeTypes}
          onNodesChange={onNodesChange}
          onEdgesChange={onEdgesChange}
          onConnect={onConnect}
          isValidConnection={isValidConnection}
          fitView
        >
          <Background />
          <Controls />
          <MiniMap />
        </ReactFlow>
      </div>
    </div>
  );
}
```

const COMPATIBLE: Record<HandleType, HandleType[]> = {
  text: ["text", "any"],
  json: ["json", "any"],
  image: ["image", "any"],
  any: ["text", "json", "image", "any"],
};

export function isValidConnection(connection: Connection): boolean {
  if (!connection.source || !connection.target) return false;
  // Allow all connections for now (strict type checking requires node lookup from state)
  return connection.source !== connection.target;
}

**web/src/hooks/use-node-execution.ts:**
```typescript
import { useState, useCallback, useRef } from "react";
import { skillsApi } from "@/lib/api";

interface ExecutionState {
  status: "idle" | "running" | "completed" | "failed";
  result: Record<string, unknown> | null;
  error: string | null;
  progress: number;
}

export function useNodeExecution(canvasId: string, nodeId: string) {
  const [state, setState] = useState<ExecutionState>({
    status: "idle", result: null, error: null, progress: 0,
  });
  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null);

  const execute = useCallback(async (skillName: string, params: Record<string, unknown>) => {
    setState({ status: "running", result: null, error: null, progress: 0 });

    try {
      const res = await skillsApi.invoke({
        skill_name: skillName,
        params,
        canvas_id: canvasId,
        node_id: nodeId,
      });

      const data = res.data;
      if (data.status === "completed") {
        setState({ status: "completed", result: data.data, error: null, progress: 1 });
        return data.data;
      }
      if (data.status === "failed") {
        setState({ status: "failed", result: null, error: data.message, progress: 0 });
        return null;
      }

      // Async: poll
      if (data.task_id) {
        return new Promise((resolve) => {
          pollRef.current = setInterval(async () => {
            try {
              const poll = await skillsApi.poll(data.task_id!);
              setState((prev) => ({ ...prev, progress: poll.data.progress }));
              if (poll.data.status === "completed") {
                clearInterval(pollRef.current!);
                pollRef.current = null;
                setState({ status: "completed", result: poll.data.data, error: null, progress: 1 });
                resolve(poll.data.data);
              } else if (poll.data.status === "failed") {
                clearInterval(pollRef.current!);
                pollRef.current = null;
                setState({ status: "failed", result: null, error: poll.data.message, progress: 0 });
                resolve(null);
              }
            } catch (err) {
              clearInterval(pollRef.current!);
              pollRef.current = null;
              setState({ status: "failed", result: null, error: String(err), progress: 0 });
              resolve(null);
            }
          }, 3000);
        });
      }
    } catch (err) {
      setState({ status: "failed", result: null, error: String(err), progress: 0 });
      return null;
    }
  }, [canvasId, nodeId]);

  const reset = useCallback(() => {
    if (pollRef.current) clearInterval(pollRef.current);
    setState({ status: "idle", result: null, error: null, progress: 0 });
  }, []);

  return { ...state, execute, reset };
}
```
