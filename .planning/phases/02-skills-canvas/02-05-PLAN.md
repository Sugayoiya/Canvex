```typescript
export const canvasApi = {
  list: (projectId: string) => api.get("/canvas/", { params: { project_id: projectId } }),
  get: (canvasId: string) => api.get(`/canvas/${canvasId}`),
  create: (data: { name: string; project_id: string; description?: string }) => api.post("/canvas/", data),
  update: (canvasId: string, data: Record<string, unknown>) => api.patch(`/canvas/${canvasId}`, data),
  delete: (canvasId: string) => api.delete(`/canvas/${canvasId}`),
  addNode: (canvasId: string, data: { node_type: string; label?: string; position_x?: number; position_y?: number; data?: Record<string, unknown> }) =>
    api.post(`/canvas/${canvasId}/nodes`, data),
  updateNode: (canvasId: string, nodeId: string, data: Record<string, unknown>) =>
    api.patch(`/canvas/${canvasId}/nodes/${nodeId}`, data),
  deleteNode: (canvasId: string, nodeId: string) => api.delete(`/canvas/${canvasId}/nodes/${nodeId}`),
  addEdge: (canvasId: string, data: { source_node_id: string; target_node_id: string; source_handle?: string; target_handle?: string }) =>
    api.post(`/canvas/${canvasId}/edges`, data),
  deleteEdge: (canvasId: string, edgeId: string) => api.delete(`/canvas/${canvasId}/edges/${edgeId}`),
};
```
