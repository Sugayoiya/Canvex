**api/app/services/ai/base.py:**
- `Message` dataclass with `role: str` ("system"|"user"|"assistant") and `content: str`.
- `LLMProviderBase` abstract class:
  - `provider_name: str`, `model_name: str`, `api_key: str`
  - `DEFAULT_TIMEOUT: int = 60` (seconds, class-level constant)
  - `MAX_RETRIES: int = 1` (class-level constant)
  - `async def generate(self, messages: list[Message], timeout: int | None = None, **kwargs) -> tuple[str, dict]`:
    1. Record `start = time.monotonic()`
    2. Use `timeout or self.DEFAULT_TIMEOUT` for the httpx/openai timeout
    3. Wrap `_generate_impl` in retry loop: try up to `1 + MAX_RETRIES` times, catching `openai.APITimeoutError`, `openai.APIConnectionError`, `openai.InternalServerError` (transient errors). On transient error, `await asyncio.sleep(2 ** attempt)` then retry.
    4. On success: compute `duration_ms`, call `await self._log_success(messages, usage, duration_ms)`
    5. On final failure: compute `duration_ms`, call `await self._log_failure(messages, str(error), duration_ms)`, then re-raise.
    6. Return `(text, usage_dict)`.
  - `async def _log_success(self, messages, usage, duration_ms)`: import and call `log_ai_call(...)` with `status="success"`. Wrap in `try/except Exception` — log error but never raise (fail-open).
  - `async def _log_failure(self, messages, error_msg, duration_ms)`: same pattern, `status="failed"`, `error_message=error_msg`. Wrap in `try/except Exception` — fail-open.
  - `async def _generate_impl(self, messages, **kwargs) -> tuple[str, dict]` — abstract.
  - Constructor: `self.model_name = model_name`, `self.api_key = api_key`.

**api/app/services/ai/ai_call_logger.py:**

Create ContextVar-based context propagation and fail-open log writer:

```python
import contextvars
import logging
import uuid
from decimal import Decimal
from typing import Any

logger = logging.getLogger(__name__)

_current_context: contextvars.ContextVar[dict[str, Any]] = contextvars.ContextVar(
    "_ai_call_context", default={}
)

def set_ai_call_context(
    trace_id: str = "",
    user_id: str = "",
    team_id: str | None = None,
    project_id: str | None = None,
    skill_execution_id: str | None = None,
) -> None:
    _current_context.set({
        "trace_id": trace_id,
        "user_id": user_id,
        "team_id": team_id,
        "project_id": project_id,
        "skill_execution_id": skill_execution_id,
    })

def get_ai_call_context() -> dict[str, Any]:
    return _current_context.get()

async def log_ai_call(
    provider: str,
    model: str,
    model_type: str = "llm",
    input_tokens: int | None = None,
    output_tokens: int | None = None,
    duration_ms: int = 0,
    status: str = "success",
    error_message: str | None = None,
    trace_id: str | None = None,
    skill_execution_id: str | None = None,
    user_id: str = "",
    team_id: str | None = None,
    project_id: str | None = None,
    input_summary: str | None = None,
    cost: Decimal | None = None,
    retry_count: int = 0,
) -> str | None:
    """Write AICallLog row. FAIL-OPEN: never raises — logs error and returns None."""
    try:
        from app.core.database import AsyncSessionLocal
        from app.models.ai_call_log import AICallLog
        log = AICallLog(
            id=str(uuid.uuid4()),
            trace_id=trace_id or "",
            skill_execution_id=skill_execution_id,
            user_id=user_id,
            team_id=team_id,
            project_id=project_id,
            provider=provider,
            model=model,
            model_type=model_type,
            input_tokens=input_tokens,
            output_tokens=output_tokens,
            duration_ms=duration_ms,
            status=status,
            error_message=error_message,
            input_summary=input_summary[:500] if input_summary else None,
            cost=cost,
            retry_count=retry_count,
        )
        async with AsyncSessionLocal() as session:
            session.add(log)
            await session.commit()
        return log.id
    except Exception:
        logger.exception("FAIL-OPEN: AICallLog write failed for %s/%s", provider, model)
        return None
```
