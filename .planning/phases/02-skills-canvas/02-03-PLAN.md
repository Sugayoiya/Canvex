**api/app/skills/llm_utils.py — shared helper (addresses review #6: structured output):**

```python
import json
import logging
import re
from typing import Any
from app.skills.context import SkillContext
from app.services.ai.base import Message
from app.services.ai.provider_manager import get_llm_provider
from app.services.ai.ai_call_logger import set_ai_call_context

logger = logging.getLogger(__name__)

async def call_llm(
    ctx: SkillContext,
    system_prompt: str,
    user_content: str,
    max_tokens: int = 2000,
    **kwargs,
) -> tuple[str, dict]:
    """Unified LLM call: sets context, gets provider, builds messages, calls generate.
    Returns (text, usage_dict). Raises on provider failure (caller handles SkillResult.failed).
    """
    set_ai_call_context(
        trace_id=ctx.trace_id,
        user_id=ctx.user_id,
        team_id=ctx.team_id,
        project_id=ctx.project_id,
    )
    provider = get_llm_provider()
    messages = [
        Message(role="system", content=system_prompt),
        Message(role="user", content=user_content),
    ]
    return await provider.generate(messages, max_tokens=max_tokens, **kwargs)


def parse_llm_json(text: str, expected_type: type = list) -> Any:
    """Parse JSON from LLM output with fallback. Strips markdown fences if present.
    Returns parsed value or None on failure.
    """
    cleaned = text.strip()
    # Strip markdown code fences
    fence_match = re.search(r"```(?:json)?\s*\n?(.*?)```", cleaned, re.DOTALL)
    if fence_match:
        cleaned = fence_match.group(1).strip()

    try:
        result = json.loads(cleaned)
        if expected_type and not isinstance(result, expected_type):
            if expected_type is list and isinstance(result, dict):
                return [result]
            return None
        return result
    except json.JSONDecodeError:
        logger.warning("Failed to parse LLM JSON output: %s...", cleaned[:100])
        return None
```
