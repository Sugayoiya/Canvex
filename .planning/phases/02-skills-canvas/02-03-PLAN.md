async def handle_llm_generate(params: dict, ctx: SkillContext) -> SkillResult:
    prompt = params.get("prompt", "")
    system_prompt = params.get("system_prompt", "你是一个专业的影视创作助手。")
    if not prompt:
        return SkillResult.failed("prompt 不能为空")

    from app.services.ai.provider_manager import get_llm_provider
    from app.services.ai.base import Message
    from app.services.ai.ai_call_logger import set_ai_call_context

    set_ai_call_context(
        trace_id=ctx.trace_id, user_id=ctx.user_id,
        team_id=ctx.team_id, project_id=ctx.project_id,
    )

    provider = get_llm_provider()
    messages = [
        Message(role="system", content=system_prompt),
        Message(role="user", content=prompt),
    ]
    max_tokens = params.get("max_tokens", 2000)
    text, usage = await provider.generate(messages, max_tokens=max_tokens)

    return SkillResult(
        status="completed",
        data={"text": text},
        message=f"生成完成 ({len(text)} 字)",
    )
```

async def handle_extract_characters(params: dict, ctx: SkillContext) -> SkillResult:
    text = params.get("text", "")
    if not text:
        return SkillResult.failed("text 不能为空")

    from app.services.ai.provider_manager import get_llm_provider
    from app.services.ai.base import Message
    from app.services.ai.ai_call_logger import set_ai_call_context

    set_ai_call_context(
        trace_id=ctx.trace_id, user_id=ctx.user_id,
        team_id=ctx.team_id, project_id=ctx.project_id,
    )

    provider = get_llm_provider()
    messages = [
        Message(role="system", content=(
            "你是专业的影视剧本分析师。请从以下文本中提取所有角色信息。"
            "返回 JSON 数组格式: [{\"name\": \"角色名\", \"description\": \"角色描述\", "
            "\"gender\": \"性别\", \"age\": \"年龄/年龄段\"}]。"
            "只返回 JSON，不要额外说明。"
        )),
        Message(role="user", content=text),
    ]
    result_text, usage = await provider.generate(messages)

    import json
    try:
        characters = json.loads(result_text)
        if not isinstance(characters, list):
            characters = [characters]
    except json.JSONDecodeError:
        characters = [{"name": "解析失败", "description": result_text[:200], "gender": "", "age": ""}]

    return SkillResult(
        status="completed",
        data={"characters": characters},
        message=f"从文本中提取到 {len(characters)} 个角色",
    )
```
