interface AdminErrorBoundaryProps {

export class AdminErrorBoundary extends Component<AdminErrorBoundaryProps, { hasError: boolean }> {
  static getDerivedStateFromError(): { hasError: true };
  componentDidCatch(error: Error): void;  // console.error for logging
  render(): ReactNode;
}

Wrap the content area below the header. The page header (h1 + subtitle + any top-level controls) stays OUTSIDE the boundary so it's always visible:

```typescript
<div style={{ display: "flex", flexDirection: "column", gap: 32 }}>
  {/* Page header stays outside — always visible even during errors */}
  <div>
    <h1>...</h1>
    <p>...</p>
  </div>
  <AdminErrorBoundary>
    {/* All content below header wrapped */}
    {/* ... existing page content (tables, cards, tabs, etc.) ... */}
  </AdminErrorBoundary>
</div>
```

Special case — Monitoring page: wrap below header but ABOVE TabBar, so a boundary error doesn't break tab navigation. Actually, since TabBar is part of the content and an error in one tab component would bubble up, the boundary should wrap the entire content area including TabBar. Individual tab components handle their own isError/isLoading via useAdminLogTable hook, so the boundary only catches truly uncaught errors.
