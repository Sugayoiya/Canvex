**1. Create web/src/components/canvas/hooks/use-node-focus.ts:**

```typescript
"use client";

import { useState, useCallback, useEffect } from "react";
import { useNodes, useReactFlow, useViewport } from "@xyflow/react";

type PanelType = "ai-generate" | "text-toolbar" | "template-menu" | null;
type PanelDirection = "above" | "below";

interface FocusState {
  focusedNodeId: string | null;
  panelType: PanelType;
  panelDirection: PanelDirection;
}

interface PanelPosition {
  x: number;
  y: number;
  width: number;
}

const GAP_ABOVE = 8;
const GAP_BELOW = 12;

export function useNodeFocus() {
  const [focusState, setFocusState] = useState<FocusState>({
    focusedNodeId: null,
    panelType: null,
    panelDirection: "below",
  });
  const nodes = useNodes();
  const viewport = useViewport();
  const { getNode } = useReactFlow();

  const focusNode = useCallback(
    (nodeId: string) => {
      const node = nodes.find((n) => n.id === nodeId);
      if (!node) return;

      const data = node.data as Record<string, unknown>;
      const hasContent = !!(
        data.result_text ||
        data.result_url ||
        data.result_video_url ||
        data.result_audio_url ||
        data.text
      );
      const nodeType = (data.nodeType as string) ?? node.type ?? "text";

      let panelType: PanelType = null;
      let panelDirection: PanelDirection = "below";

      if (!hasContent) {
        panelType = "ai-generate";
        panelDirection = "below";
      } else if (nodeType === "text" || nodeType === "text-input" || nodeType === "llm-generate" || nodeType === "source-text" || nodeType === "ai-text-generate" || nodeType === "prompt-input" || nodeType === "extract" || nodeType === "output") {
        panelType = "text-toolbar";
        panelDirection = "above";
      } else {
        panelType = "template-menu";
        panelDirection = "above";
      }

      setFocusState({ focusedNodeId: nodeId, panelType, panelDirection });
    },
    [nodes],
  );

  const blurNode = useCallback(() => {
    setFocusState({ focusedNodeId: null, panelType: null, panelDirection: "below" });
  }, []);

  const getPanelPosition = useCallback((): PanelPosition | null => {
    if (!focusState.focusedNodeId) return null;
    const node = getNode(focusState.focusedNodeId);
    if (!node) return null;

    const nodeWidth = node.measured?.width ?? 300;
    const nodeHeight = node.measured?.height ?? 200;

    const screenX = node.position.x * viewport.zoom + viewport.x;
    const screenY = node.position.y * viewport.zoom + viewport.y;
    const scaledHeight = nodeHeight * viewport.zoom;

    if (focusState.panelDirection === "above") {
      return {
        x: screenX,
        y: screenY - GAP_ABOVE,
        width: nodeWidth * viewport.zoom,
      };
    } else {
      return {
        x: screenX,
        y: screenY + scaledHeight + GAP_BELOW,
        width: nodeWidth * viewport.zoom,
      };
    }
  }, [focusState, viewport, getNode]);

  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === "Escape") blurNode();
    };
    window.addEventListener("keydown", handleKeyDown);
    return () => window.removeEventListener("keydown", handleKeyDown);
  }, [blurNode]);

  return {
    ...focusState,
    focusNode,
    blurNode,
    getPanelPosition,
  };
}
```

**2. Create web/src/components/canvas/hooks/use-prompt-builder.ts:**

```typescript
"use client";

import { useMemo } from "react";
import { useNodes } from "@xyflow/react";
import { useUpstreamData } from "./use-upstream-data";

interface PromptBuilderResult {
  finalPrompt: string;
  upstreamImages: string[];
  upstreamVideos: string[];
  upstreamAudios: string[];
  hasUpstreamContent: boolean;
}

export function usePromptBuilder(nodeId: string): PromptBuilderResult {
  const upstream = useUpstreamData(nodeId);
  const nodes = useNodes();

  return useMemo(() => {
    const node = nodes.find((n) => n.id === nodeId);
    const data = (node?.data ?? {}) as Record<string, unknown>;
    const config = (data.config ?? {}) as Record<string, unknown>;

    const parts: string[] = [];

    // 1. Hidden prompt (from template system)
    const hiddenPrompt = config.hidden_prompt as string | undefined;
    if (hiddenPrompt) parts.push(hiddenPrompt);

    // 2. Node's own description/prompt
    const nodeText = (data.text as string) || (data.prompt as string) || "";
    if (nodeText) parts.push(nodeText);

    // 3. Upstream text content
    if (upstream.text.length > 0) parts.push(upstream.text.join("\n"));

    return {
      finalPrompt: parts.join("\n\n"),
      upstreamImages: upstream.imageUrl,
      upstreamVideos: upstream.videoUrl,
      upstreamAudios: upstream.audioUrl,
      hasUpstreamContent: upstream.text.length > 0 || upstream.imageUrl.length > 0 || upstream.videoUrl.length > 0,
    };
  }, [nodeId, upstream, nodes]);
}
```

**1. Create web/src/components/canvas/nodes/text-node.tsx:**

Minimal text card using NodeShell. Empty state: `FileText` icon (28px, `--cv4-text-disabled`) + hints. Content state: left-border text preview with title detection.

```typescript
"use client";
import { useCallback } from "react";
import { type NodeProps, useReactFlow } from "@xyflow/react";
import { FileText, PenLine } from "lucide-react";
import { NodeShell } from "./shared/node-shell";
import { StatusIndicator } from "./shared/status-indicator";
import { useNodePersistence } from "../hooks/use-node-persistence";
import { useNodeExecution } from "../hooks/use-node-execution";

const HINTS = ["✏ 自己编写内容", "▶ 文生视频", "🖼 图片反推提示词", "🎵 文字生音乐"];

export function TextNode({ id, data }: NodeProps) {
  const d = data as Record<string, unknown>;
  const status = (d.status as string) ?? "idle";
  const text = (d.result_text as string) || (d.text as string) || "";
  const hasContent = !!text;

  const { setNodes } = useReactFlow();
  const persistence = useNodePersistence(id);
  const handleComplete = useCallback((resultData: any) => {
    persistence.cancelPending();
    setNodes((nds) => nds.map((n) =>
      n.id === id ? { ...n, data: { ...n.data, result_text: typeof resultData === "string" ? resultData : resultData?.text ?? resultData?.result ?? "", result_data: typeof resultData === "object" ? resultData : null } } : n
    ));
  }, [id, setNodes, persistence]);
  useNodeExecution(id, handleComplete);

  return (
    <NodeShell nodeId={id} nodeType="text" footer={hasContent ? (
      <div className="flex items-center" style={{ padding: "8px 16px" }}>
        <span className="flex-1" />
        <StatusIndicator status={status} />
        <PenLine size={14} style={{ color: "var(--cv4-text-disabled)", marginLeft: 8 }} />
      </div>
    ) : undefined}>
      {hasContent ? (
        <div className="flex gap-3" style={{ padding: "12px 16px" }}>
          <div style={{ width: 3, borderRadius: 2, background: "var(--cv4-text-disabled)", opacity: 0.6, flexShrink: 0 }} />
          <div className="flex flex-col gap-2 overflow-hidden">
            <p className="font-['Manrope'] text-[15px] font-bold leading-[1.6] line-clamp-1" style={{ color: "var(--cv4-text-primary)" }}>
              {text.split("\n")[0]}
            </p>
            <p className="font-['Manrope'] text-xs leading-[1.6] line-clamp-4" style={{ color: "var(--cv4-text-secondary)" }}>
              {text.split("\n").slice(1).join("\n")}
            </p>
          </div>
        </div>
      ) : (
        <>
          <div className="flex items-center justify-center" style={{ height: 100, background: "var(--cv4-surface-secondary)" }}>
            <FileText size={28} style={{ color: "var(--cv4-text-disabled)" }} />
          </div>
          <div className="flex flex-col gap-2" style={{ padding: "12px 16px" }}>
            {HINTS.map((h) => (
              <span key={h} className="font-['Manrope'] text-xs" style={{ color: "var(--cv4-text-disabled)" }}>{h}</span>
            ))}
          </div>
        </>
      )}
    </NodeShell>
  );
}
```
