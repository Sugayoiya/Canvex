```python
"""GeminiVideoProvider — Veo API wrapper for video generation."""
import logging
import os
import time
import uuid
from typing import Any

logger = logging.getLogger(__name__)


class GeminiVideoProvider:
    """Generate videos using Gemini Veo API (google-genai SDK)."""

    def __init__(self, api_key: str, model: str = "veo-2.0-generate-001"):
        self.api_key = api_key
        self.model = model

    async def generate_video(
        self,
        prompt: str,
        *,
        image_bytes: bytes | None = None,
        aspect_ratio: str = "16:9",
        duration_seconds: int = 5,
    ) -> dict[str, Any]:
        from google import genai
        from google.genai import types

        client = genai.Client(api_key=self.api_key)

        config = types.GenerateVideosConfig(
            aspect_ratio=aspect_ratio,
            number_of_videos=1,
        )

        try:
            kwargs: dict[str, Any] = {
                "model": self.model,
                "prompt": prompt,
                "config": config,
            }
            if image_bytes:
                kwargs["image"] = types.Image(image_bytes=image_bytes)

            operation = await client.aio.models.generate_videos(**kwargs)

            # Poll for completion (Veo returns a long-running operation)
            max_polls = 60
            poll_interval = 5
            for _ in range(max_polls):
                operation = await client.aio.operations.get(operation)
                if operation.done:
                    break
                import asyncio
                await asyncio.sleep(poll_interval)

            if not operation.done:
                raise TimeoutError(f"Video generation timed out after {max_polls * poll_interval}s")

            # Extract video from response
            result = operation.response
            if hasattr(result, "generated_videos") and result.generated_videos:
                video = result.generated_videos[0]
                video_data = video.video.video_bytes if hasattr(video, "video") else None

                if video_data:
                    upload_dir = os.environ.get("UPLOAD_DIR", "uploads")
                    os.makedirs(f"{upload_dir}/generated", exist_ok=True)
                    filename = f"gen_{uuid.uuid4().hex[:12]}.mp4"
                    filepath = f"{upload_dir}/generated/{filename}"
                    with open(filepath, "wb") as f:
                        f.write(video_data)
                    return {
                        "url": f"/api/v1/files/generated/{filename}",
                        "filename": filename,
                        "duration_seconds": duration_seconds,
                    }

            raise ValueError("No video generated in response")

        except ImportError:
            raise ValueError("google-genai SDK not installed or incompatible version")
        except Exception as e:
            error_msg = str(e)
            if "not found" in error_msg.lower() or "permission" in error_msg.lower():
                raise ValueError(f"Video generation model not available: {error_msg[:200]}")
            raise
```

```python
"""video.generate_video — Generate videos via Gemini Veo API."""
import logging
import time
from typing import Any

from app.skills.descriptor import SkillDescriptor, SkillCategory, SkillResult
from app.skills.context import SkillContext
from app.skills.registry import skill_registry

logger = logging.getLogger(__name__)

descriptor = SkillDescriptor(
    name="video.generate_video",
    display_name="AI 视频生成",
    description="使用 Gemini Veo API 根据提示词或首帧图片生成视频，返回视频 URL。",
    category=SkillCategory.VIDEO,
    triggers=["生成视频", "AI 视频", "generate video", "text to video", "image to video"],
    execution_mode="async_celery",
    celery_queue="media_processing",
    estimated_duration="long",
    requires_project=False,
    input_schema={
        "type": "object",
        "properties": {
            "prompt": {"type": "string", "description": "视频生成提示词"},
            "image_url": {"type": "string", "description": "首帧图片 URL (可选，用于图生视频)"},
            "aspect_ratio": {"type": "string", "description": "画面比例", "default": "16:9"},
            "duration_seconds": {"type": "integer", "description": "视频时长（秒）", "default": 5},
            "model": {"type": "string", "description": "Veo 模型名称", "default": "veo-2.0-generate-001"},
        },
        "required": ["prompt"],
    },
    output_schema={
        "type": "object",
        "properties": {
            "url": {"type": "string", "description": "生成视频的访问 URL"},
            "filename": {"type": "string", "description": "文件名"},
            "duration_seconds": {"type": "integer"},
        },
    },
)


async def handle_generate_video(params: dict[str, Any], ctx: SkillContext) -> SkillResult:
    prompt = params.get("prompt", "")
    if not prompt:
        return SkillResult.failed("prompt 不能为空")

    aspect_ratio = params.get("aspect_ratio", "16:9")
    duration_seconds = params.get("duration_seconds", 5)
    model = params.get("model", "veo-2.0-generate-001")
    image_url = params.get("image_url")

    logger.info(
        "video.generate_video invoked (trace=%s, model=%s, ratio=%s, duration=%ds)",
        ctx.trace_id, model, aspect_ratio, duration_seconds,
    )

    from app.services.ai.ai_call_logger import set_ai_call_context, log_ai_call

    set_ai_call_context(
        trace_id=ctx.trace_id, user_id=ctx.user_id,
        team_id=ctx.team_id, project_id=ctx.project_id,
    )

    from app.core.config import settings
    api_key = settings.GEMINI_API_KEY
    if not api_key:
        return SkillResult.failed("GEMINI_API_KEY 未配置")

    # Load first-frame image if provided
    image_bytes = None
    if image_url:
        try:
            import os
            upload_dir = os.environ.get("UPLOAD_DIR", "uploads")
            local_path = image_url.replace("/api/v1/files/", f"{upload_dir}/")
            if os.path.exists(local_path):
                with open(local_path, "rb") as f:
                    image_bytes = f.read()
        except Exception as e:
            logger.warning("Failed to load first-frame image: %s", e)

    start = time.monotonic()
    try:
        from app.services.ai.model_providers.gemini_video import GeminiVideoProvider

        provider = GeminiVideoProvider(api_key=api_key, model=model)
        result = await provider.generate_video(
            prompt,
            image_bytes=image_bytes,
            aspect_ratio=aspect_ratio,
            duration_seconds=duration_seconds,
        )
        duration_ms = int((time.monotonic() - start) * 1000)

        await log_ai_call(
            provider="gemini", model=model, model_type="video",
            status="success", duration_ms=duration_ms,
        )

        return SkillResult(
            status="completed",
            data=result,
            artifacts=[{"type": "video", "url": result["url"]}],
            message="视频生成完成",
        )
    except ValueError as e:
        duration_ms = int((time.monotonic() - start) * 1000)
        await log_ai_call(
            provider="gemini", model=model, model_type="video",
            status="error", error_message=str(e)[:200], duration_ms=duration_ms,
        )
        return SkillResult.failed(f"视频生成失败: {str(e)[:200]}")
    except Exception as e:
        duration_ms = int((time.monotonic() - start) * 1000)
        logger.exception("video.generate_video failed")
        await log_ai_call(
            provider="gemini", model=model, model_type="video",
            status="error", error_message=str(e)[:200], duration_ms=duration_ms,
        )
        return SkillResult.failed(f"视频生成失败: {str(e)[:200]}")


def register_generate_video_skill():
    skill_registry.register(descriptor, handle_generate_video)
```
