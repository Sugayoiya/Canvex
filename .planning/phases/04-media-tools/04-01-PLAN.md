"use client";

import { Handle, Position } from "@xyflow/react";
import { FileText, Image, PlayCircle, Music, GripHorizontal } from "lucide-react";
import type { ReactNode } from "react";

const NODE_ICONS: Record<string, typeof FileText> = {
  text: FileText,
  image: Image,
  video: PlayCircle,
  audio: Music,
};

const NODE_TYPE_LABELS: Record<string, string> = {
  text: "文本",
  image: "图片",
  video: "视频",
  audio: "音频",
};

const NODE_WIDTHS: Record<string, number> = {
  text: 280,
  image: 340,
  video: 340,
  audio: 280,
};

interface NodeShellProps {
  nodeId: string;
  nodeType: "text" | "image" | "video" | "audio";
  isFocused?: boolean;
  children: ReactNode;
  footer?: ReactNode;
}

export function NodeShell({ nodeId, nodeType, isFocused, children, footer }: NodeShellProps) {
  const Icon = NODE_ICONS[nodeType] ?? FileText;
  const label = NODE_TYPE_LABELS[nodeType] ?? nodeType;
  const width = NODE_WIDTHS[nodeType] ?? 280;
  const idSuffix = nodeId.slice(-4);

  return (
    <div
      className="relative transition-all duration-[120ms]"
      style={{
        width: `${width}px`,
        background: "var(--cv4-surface-primary)",
        borderRadius: "var(--cv4-radius-node)",
        border: isFocused
          ? "1.5px solid var(--cv4-border-focused)"
          : "1px solid var(--cv4-border-default)",
        boxShadow: isFocused ? "var(--cv4-shadow-lg)" : "var(--cv4-shadow-md)",
      }}
    >
      <Handle
        type="target"
        position={Position.Left}
        id="input"
        className="!h-3 !w-3 !border-2 !bg-transparent"
        style={{ borderColor: "var(--cv4-border-default)" }}
      />

      {/* Header */}
      <div
        className="flex items-center gap-2"
        style={{ padding: "12px 16px" }}
      >
        <Icon size={14} style={{ color: "var(--cv4-text-muted)" }} />
        <span
          className="font-['Space_Grotesk'] text-xs"
          style={{ color: "var(--cv4-text-secondary)" }}
        >
          {label} {idSuffix}
        </span>
        <span className="flex-1" />
        <GripHorizontal size={14} style={{ color: "var(--cv4-text-disabled)" }} />
      </div>

      {/* Content area */}
      {children}

      {/* Footer (if provided) */}
      {footer}

      <Handle
        type="source"

        id="output"
        className="!h-3 !w-3 !border-2 !bg-transparent"

  );

"use client";

import { Loader2, CheckCircle2, XCircle, Clock, AlertTriangle, Ban } from "lucide-react";
import type { ReactNode } from "react";

type NodeStatus = "idle" | "queued" | "running" | "completed" | "failed" | "timeout" | "blocked";

const STATUS_CONFIG: Record<NodeStatus, { icon: ReactNode; label: string; color: string }> = {
  idle:      { icon: null, label: "", color: "var(--cv4-text-disabled)" },
  queued:    { icon: <Clock size={12} className="animate-pulse" />, label: "排队中", color: "#EAB308" },
  running:   { icon: <Loader2 size={12} className="animate-spin" />, label: "执行中", color: "#3B82F6" },
  completed: { icon: <CheckCircle2 size={12} />, label: "完成", color: "#22C55E" },
  failed:    { icon: <XCircle size={12} />, label: "失败", color: "#EF4444" },
  timeout:   { icon: <AlertTriangle size={12} />, label: "超时", color: "#F97316" },
  blocked:   { icon: <Ban size={12} />, label: "已阻止", color: "#EF4444" },
};

interface StatusIndicatorProps {
  status: string;
}

export function StatusIndicator({ status }: StatusIndicatorProps) {
  const config = STATUS_CONFIG[status as NodeStatus] ?? STATUS_CONFIG.idle;
  if (!config.icon) return null;

  return (
    <div className="flex items-center gap-1" style={{ color: config.color }}>
      {config.icon}
      <span className="font-['Space_Grotesk'] text-[9px] font-bold uppercase tracking-wider">
        {config.label}
      </span>
    </div>
  );
}
