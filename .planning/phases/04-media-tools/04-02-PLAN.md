**3. Create api/app/services/quota_service.py:**

Implement `QuotaService` with:
- `check_quota(user_id, team_id=None) -> QuotaCheckResult` — checks user quota then team quota. Returns `QuotaCheckResult(allowed=True)` or `QuotaCheckResult(allowed=False, reason="...", error_code="USER_QUOTA_EXCEEDED"|"TEAM_QUOTA_EXCEEDED")`.
- Lazy reset: if `last_reset_date.month != now.month`, reset `current_month_usage` to 0. If `last_reset_date.date() != now.date()`, reset `current_day_calls` to 0.
- `update_usage(user_id, team_id, cost: Decimal)` — increment `current_month_usage` and `current_day_calls` after successful execution.
- NULL limits mean unlimited — skip check.

```python
from dataclasses import dataclass
from decimal import Decimal
from datetime import datetime
import logging
from sqlalchemy import select
from app.core.database import AsyncSessionLocal
from app.models.quota import UserQuota, TeamQuota

logger = logging.getLogger(__name__)

@dataclass
class QuotaCheckResult:
    allowed: bool
    reason: str = ""
    error_code: str = ""

class QuotaService:
    @staticmethod
    async def check_quota(user_id: str, team_id: str | None = None) -> QuotaCheckResult:
        try:
            async with AsyncSessionLocal() as session:
                # Check user quota
                stmt = select(UserQuota).where(UserQuota.user_id == user_id)
                uq = (await session.execute(stmt)).scalar_one_or_none()
                if uq:
                    QuotaService._lazy_reset(uq)
                    if uq.monthly_credit_limit is not None and uq.current_month_usage >= uq.monthly_credit_limit:
                        return QuotaCheckResult(False, "用户月度额度已用完", "USER_QUOTA_EXCEEDED")
                    if uq.daily_call_limit is not None and uq.current_day_calls >= uq.daily_call_limit:
                        return QuotaCheckResult(False, "用户每日调用次数已用完", "USER_DAILY_EXCEEDED")
                    await session.commit()

                # Check team quota
                if team_id:
                    stmt = select(TeamQuota).where(TeamQuota.team_id == team_id)
                    tq = (await session.execute(stmt)).scalar_one_or_none()
                    if tq:
                        QuotaService._lazy_reset(tq)
                        if tq.monthly_credit_limit is not None and tq.current_month_usage >= tq.monthly_credit_limit:
                            return QuotaCheckResult(False, "团队月度额度已用完", "TEAM_QUOTA_EXCEEDED")
                        if tq.daily_call_limit is not None and tq.current_day_calls >= tq.daily_call_limit:
                            return QuotaCheckResult(False, "团队每日调用次数已用完", "TEAM_DAILY_EXCEEDED")
                        await session.commit()
        except Exception:
            logger.exception("Quota check failed — fail-open, allowing execution")
            return QuotaCheckResult(allowed=True)

        return QuotaCheckResult(allowed=True)

    @staticmethod
    def _lazy_reset(quota: UserQuota | TeamQuota) -> None:
        now = datetime.utcnow()
        if quota.last_reset_date.month != now.month or quota.last_reset_date.year != now.year:
            quota.current_month_usage = Decimal("0")
            quota.current_day_calls = 0
            quota.last_reset_date = now
        elif quota.last_reset_date.date() != now.date():
            quota.current_day_calls = 0
            quota.last_reset_date = now

    @staticmethod
    async def update_usage(user_id: str, team_id: str | None, cost: Decimal) -> None:
        try:
            async with AsyncSessionLocal() as session:
                stmt = select(UserQuota).where(UserQuota.user_id == user_id)
                uq = (await session.execute(stmt)).scalar_one_or_none()
                if uq:
                    QuotaService._lazy_reset(uq)
                    uq.current_month_usage += cost
                    uq.current_day_calls += 1

                if team_id:
                    stmt = select(TeamQuota).where(TeamQuota.team_id == team_id)
                    tq = (await session.execute(stmt)).scalar_one_or_none()
                    if tq:
                        QuotaService._lazy_reset(tq)
                        tq.current_month_usage += cost
                        tq.current_day_calls += 1

                await session.commit()
        except Exception:
            logger.exception("Quota usage update failed — non-blocking")
```

from decimal import Decimal
from pydantic import BaseModel

class QuotaResponse(BaseModel):
    user_id: str | None = None
    team_id: str | None = None
    monthly_credit_limit: Decimal | None = None
    daily_call_limit: int | None = None
    current_month_usage: Decimal = Decimal("0")
    current_day_calls: int = 0

class QuotaUpdate(BaseModel):
    monthly_credit_limit: Decimal | None = None
    daily_call_limit: int | None = None

class QuotaCheckResponse(BaseModel):
    allowed: bool
    reason: str = ""
    error_code: str = ""

class CostEstimateRequest(BaseModel):
    model: str
    estimated_tokens: int = 1000

class CostEstimateResponse(BaseModel):
    estimated_cost: Decimal
    currency: str = "credits"

@router.get("/quota/check", response_model=QuotaCheckResponse)
async def check_my_quota(
    current_user=Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    result = await QuotaService.check_quota(current_user.id, getattr(current_user, "current_team_id", None))
    return QuotaCheckResponse(allowed=result.allowed, reason=result.reason, error_code=result.error_code)

@router.get("/quota/user/{user_id}", response_model=QuotaResponse)
async def get_user_quota(user_id: str, current_user=Depends(get_current_user), db: AsyncSession = Depends(get_db)):
    stmt = select(UserQuota).where(UserQuota.user_id == user_id)
    quota = (await db.execute(stmt)).scalar_one_or_none()
    if not quota:
        return QuotaResponse(user_id=user_id)
    return QuotaResponse(
        user_id=quota.user_id,
        monthly_credit_limit=quota.monthly_credit_limit,
        daily_call_limit=quota.daily_call_limit,
        current_month_usage=quota.current_month_usage,
        current_day_calls=quota.current_day_calls,
    )

@router.put("/quota/user/{user_id}", response_model=QuotaResponse)
async def set_user_quota(user_id: str, body: QuotaUpdate, current_user=Depends(get_current_user), db: AsyncSession = Depends(get_db)):
    stmt = select(UserQuota).where(UserQuota.user_id == user_id)
    quota = (await db.execute(stmt)).scalar_one_or_none()
