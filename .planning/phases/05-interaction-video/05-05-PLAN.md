```typescript
const { data: statsData } = useQuery({

    end_date: dateRange.end,
    project_id: selectedProjectId ?? undefined,

const { data: timeseriesData } = useQuery({

  queryFn: () => billingApi.usageTimeseries({

    end_date: dateRange.end,
    granularity,
    project_id: selectedProjectId ?? undefined,
  }).then(r => r.data),
});
```
