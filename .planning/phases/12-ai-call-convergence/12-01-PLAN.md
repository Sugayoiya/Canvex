    **Error redaction function** (addresses review concern: [HIGH] error message data sensitivity):
    ```python
    _SENSITIVE_PATTERNS = [
        re.compile(r'(sk-[a-zA-Z0-9]{20,})'),       # OpenAI-style keys
        re.compile(r'(AIza[a-zA-Z0-9_-]{35,})'),     # Google API keys
        re.compile(r'(Bearer\s+[a-zA-Z0-9._-]{20,})'), # Auth tokens
        re.compile(r'(api[_-]?key[=:]\s*\S{10,})', re.IGNORECASE),  # Generic api_key=
        re.compile(r'(https?://[^@\s]*:[^@\s]*@)'),  # URLs with credentials
    ]

    def _redact_error_message(message: str, max_length: int = 200) -> str:
        """Strip sensitive patterns (API keys, tokens, URLs with creds) before Redis storage."""
        redacted = message
        for pattern in _SENSITIVE_PATTERNS:
            redacted = pattern.sub('[REDACTED]', redacted)
        return redacted[:max_length]
    ```

    **Redis degraded fallback** (addresses review concern: [HIGH] Redis degraded fallback):
    All Redis operations wrapped in try/except. On `RedisConnectionError`, `RedisTimeoutError`, or `RedisError`, log warning and return safe defaults:

    **Redis key naming** (per research discretion, `canvex:` namespace):

    **Methods:**

    `async def get_health(self, key_id: str) -> dict`:
    ```python
    try:
        r = await self._get_redis()
        data = await r.hgetall(f"canvex:key_health:{key_id}")
        error_count = int(data.get("error_count", 0))
        return {
            "error_count": error_count,
            "last_used_at": data.get("last_used_at"),
            "last_error_type": data.get("last_error_type"),
            "is_healthy": error_count < MAX_KEY_ERRORS,
        }
    except (RedisConnectionError, RedisTimeoutError, RedisError) as e:
        logger.warning("redis_degraded_get_health", key_id=key_id, error=str(e))
        return {"error_count": 0, "last_used_at": None, "last_error_type": None, "is_healthy": True}
    ```

    `async def report_success(self, key_id: str)`:
    Use pipeline (atomic). On Redis error, log and return (no-op).
    ```python
    try:
        r = await self._get_redis()
        now = datetime.now(timezone.utc).isoformat()
        hour_key = f"canvex:key_usage:{key_id}:{datetime.now(timezone.utc).strftime('%Y%m%d%H')}"
        pipe = r.pipeline()
    ```

    `async def report_error(self, key_id: str, error_type: str, error_message: str)`:
    Use atomic pipeline (addresses review concern: [HIGH] race conditions — pipeline executes atomically). Redact error message before storage.
    ```python
    try:
        r = await self._get_redis()
        now = datetime.now(timezone.utc).isoformat()
        redacted_msg = _redact_error_message(error_message)
        error_entry = json.dumps({"type": error_type, "message": redacted_msg, "at": now})
        pipe = r.pipeline()
        pipe.hincrby(f"canvex:key_health:{key_id}", "error_count", 1)
        pipe.hset(f"canvex:key_health:{key_id}", mapping={
            "last_used_at": now,
            "last_error_type": error_type,
            "last_error_at": now,
        })
        pipe.lpush(f"canvex:key_errors:{key_id}", error_entry)
        pipe.ltrim(f"canvex:key_errors:{key_id}", 0, 19)
        await pipe.execute()
        logger.info("key_health_error_reported", key_id=key_id, error_type=error_type)
    except (RedisConnectionError, RedisTimeoutError, RedisError) as e:
        logger.warning("redis_degraded_report_error", key_id=key_id, error=str(e))
    ```

    Create `api/app/services/ai/credential_cache.py` implementing Redis credential resolution cache.

    **CRITICAL SECURITY DESIGN** (addresses review concern: [HIGH] credential security in Redis):
    Cache stores ONLY key selection metadata — `{"key_id": str, "config_id": str, "provider_name": str, "owner_type": str, "owner_desc": str}`. It does NOT store decrypted API keys. On cache hit, the caller does a single-row DB lookup by `key_id` to get the encrypted key and decrypt it. This trades the expensive chain-walk query (team→personal→system with joins, 3+ DB queries) for a single primary-key lookup (1 query).

    **Imports and constants:**
    ```python
    import json
    import random
    import logging
    from typing import Any
    import redis.asyncio as aioredis
    from redis.exceptions import ConnectionError as RedisConnectionError, TimeoutError as RedisTimeoutError, RedisError
    from app.core.config import settings

    logger = logging.getLogger(__name__)
    CRED_CACHE_TTL = 60  # seconds (base)
    CRED_CACHE_JITTER = 5  # ±5 seconds jitter (actual TTL: 55-65s)
    LOCK_TTL = 5  # seconds for single-flight lock
    ```

    **Redis key naming:**
    - Cache: `canvex:cred:{provider}:{owner_type}:{owner_id_or_system}` — Value: JSON with key selection metadata (no API keys!)
    - Lock: `canvex:cred_lock:{provider}:{owner_type}:{owner_id_or_system}` — Lock for single-flight population

    **Class `CredentialCache`:**
    - `__init__(self)`: `self._redis: aioredis.Redis | None = None`
    - `async def _get_redis(self) -> aioredis.Redis`: Lazy init from `settings.REDIS_URL`, `decode_responses=True`.
    - `async def close(self)`: Close Redis connection.

    **Methods:**

    `async def get_cached(self, provider: str, owner_chain: list[tuple[str, str | None]]) -> dict | None`:
    Check Redis for each level in the credential chain. Returns metadata dict (key_id, config_id, etc.) or None.
    ```python
    try:
        r = await self._get_redis()
        for owner_type, owner_id in owner_chain:
            cache_key = f"canvex:cred:{provider}:{owner_type}:{owner_id or 'system'}"
            data = await r.get(cache_key)
            if data:
                logger.debug("cred_cache_hit", provider=provider, owner_type=owner_type)
                return json.loads(data)
        logger.debug("cred_cache_miss", provider=provider)
        return None
    except (RedisConnectionError, RedisTimeoutError, RedisError) as e:
        logger.warning("redis_degraded_get_cached", provider=provider, error=str(e))
        return None  # cache miss → forces DB query (safe fallback)
    ```

    `async def set_cached(self, provider: str, owner_type: str, owner_id: str | None, value: dict)`:
    SETEX with jittered TTL. Value MUST NOT contain decrypted API keys — only metadata.
    ```python
    try:
        r = await self._get_redis()
        cache_key = f"canvex:cred:{provider}:{owner_type}:{owner_id or 'system'}"
        ttl = CRED_CACHE_TTL + random.randint(-CRED_CACHE_JITTER, CRED_CACHE_JITTER)
        await r.setex(cache_key, ttl, json.dumps(value))
        logger.debug("cred_cache_set", provider=provider, owner_type=owner_type, ttl=ttl)
    except (RedisConnectionError, RedisTimeoutError, RedisError) as e:
        logger.warning("redis_degraded_set_cached", provider=provider, error=str(e))
    ```

    **Single-flight lock** (addresses review concern: [MEDIUM] cache stampede):
    `async def acquire_populate_lock(self, provider: str, owner_type: str, owner_id: str | None) -> bool`:
    ```python
    try:
        r = await self._get_redis()
        lock_key = f"canvex:cred_lock:{provider}:{owner_type}:{owner_id or 'system'}"
        acquired = await r.set(lock_key, "1", ex=LOCK_TTL, nx=True)
        if acquired:
            logger.debug("cred_cache_lock_acquired", provider=provider, owner_type=owner_type)
        return bool(acquired)
    except (RedisConnectionError, RedisTimeoutError, RedisError):
        return True  # On Redis error, allow population (degrade to no-lock)
    ```

    `async def release_populate_lock(self, provider: str, owner_type: str, owner_id: str | None)`:
    ```python
    try:
        r = await self._get_redis()
        lock_key = f"canvex:cred_lock:{provider}:{owner_type}:{owner_id or 'system'}"
        await r.delete(lock_key)
    except (RedisConnectionError, RedisTimeoutError, RedisError):
        pass  # lock will auto-expire via TTL
    ```

    `async def invalidate(self, provider: str, owner_type: str | None = None, owner_id: str | None = None)`:
    If owner_type given, delete specific key. If only provider given, use SCAN to delete all `canvex:cred:{provider}:*`.
    ```python
    try:
        r = await self._get_redis()
        if owner_type:
            cache_key = f"canvex:cred:{provider}:{owner_type}:{owner_id or 'system'}"
            await r.delete(cache_key)
        else:
            pattern = f"canvex:cred:{provider}:*"
            cursor = 0
            while True:
                cursor, keys = await r.scan(cursor, match=pattern, count=100)
                if keys:
                    await r.delete(*keys)
                if cursor == 0:
                    break
        logger.info("cred_cache_invalidated", provider=provider, owner_type=owner_type)
    except (RedisConnectionError, RedisTimeoutError, RedisError) as e:
        logger.warning("redis_degraded_invalidate", provider=provider, error=str(e))
    ```

    ```python
        global _cache

            _cache = CredentialCache()
        return _cache
    ```
