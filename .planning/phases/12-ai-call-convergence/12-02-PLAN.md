    **D. Add get_provider_with_retry():**

    ```python
    async def get_provider_with_retry(
        self, provider: str, model: str | None = None,
        *, team_id: str | None = None, user_id: str | None = None,
        db: AsyncSession | None = None, max_retries: int = 2,
    ) -> tuple[AIProviderBase, str, str]:
        """
        Calls get_provider() with automatic key rotation on failure.
        max_retries=2 for LLM (idempotent), =1 for image/video (cost-sensitive).
        Addresses review concern: [HIGH] retry/idempotency gap.
        """
        last_error = None
        for attempt in range(max_retries + 1):
            try:
                return await self.get_provider(
                    provider, model, team_id=team_id, user_id=user_id, db=db,
                )
            except ValueError:
                raise  # No healthy keys — don't retry
            except Exception as e:
                last_error = e
                key_id = _current_key_id_var.get(None)
                if key_id:
                    await get_key_health_manager().report_error(
                        key_id, type(e).__name__, str(e)
                    )
                logger.warning("provider_retry",
                    attempt=attempt + 1, max_retries=max_retries,
                    provider=provider, error=str(e))
                if attempt == max_retries:
                    raise
        raise last_error  # unreachable, but satisfies type checker
    ```
