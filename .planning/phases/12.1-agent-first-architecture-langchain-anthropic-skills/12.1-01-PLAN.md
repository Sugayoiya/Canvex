   ```python
   from dataclasses import dataclass

   @dataclass
   class ToolContext:
       project_id: str
       user_id: str
       team_id: str | None = None
       canvas_id: str | None = None
       episode_id: str | None = None

   _tool_ctx_var: contextvars.ContextVar[ToolContext | None] = contextvars.ContextVar("_tool_ctx_var", default=None)

   def set_tool_context(project_id: str, user_id: str, team_id: str | None = None, canvas_id: str | None = None, episode_id: str | None = None) -> None:

   def get_tool_context() -> ToolContext:
       ctx = _tool_ctx_var.get()
       if ctx is None:
           raise RuntimeError("Tool context not set — call set_tool_context() before agent run")
       return ctx

   def clear_tool_context() -> None:
       _tool_ctx_var.set(None)
   ```

1. `generate_image(prompt, aspect_ratio="16:9", model="imagen-4.0-generate-001")`:
   ```python
   @tool(args_schema=GenerateImageInput)
   async def generate_image(prompt: str, aspect_ratio: str = "16:9", model: str = "imagen-4.0-generate-001") -> str:
       """Generate an image using AI. Returns JSON with url and filename."""
       from app.agent.tool_context import get_tool_context
       from app.services.ai.provider_manager import get_provider_manager
       from app.services.ai.key_health import get_key_health_manager
       from app.services.ai.ai_call_logger import set_ai_call_context, log_ai_call
       import time

       ctx = get_tool_context()
       set_ai_call_context(trace_id=None, user_id=ctx.user_id, team_id=ctx.team_id, project_id=ctx.project_id)

       pm = get_provider_manager()
       provider, _owner, key_id = await pm.get_provider("gemini", team_id=ctx.team_id, user_id=ctx.user_id)

       start = time.monotonic()
       result = await provider.generate_image(prompt, aspect_ratio=aspect_ratio, model=model)
       duration_ms = int((time.monotonic() - start) * 1000)
       await get_key_health_manager().report_success(key_id)
       await log_ai_call(provider="gemini", model=model, model_type="image", status="success", duration_ms=duration_ms)
       return json.dumps(result, ensure_ascii=False)
   ```
   Error handling: catch `ContentBlockedError` → return error JSON with "内容安全策略拦截"; catch general Exception → return error JSON. Always report_error to key_health on failure.

**api/app/agent/tools/skill_tools.py** — 2 skill reader @tool functions:

1. `read_skill(name)` — Read SKILL.md body (L2 loading):
   ```python
   @tool(args_schema=ReadSkillInput)
   async def read_skill(name: str) -> str:
       """Read the detailed instructions for a specific skill. Call this before executing a complex workflow."""
       from pathlib import Path
       skills_dir = Path(__file__).parent.parent / "skills"
       skill_md = skills_dir / name / "SKILL.md"
       if not skill_md.exists():
           return json.dumps({"error": f"Skill '{name}' not found"})
       content = skill_md.read_text(encoding="utf-8")
       if content.startswith("---"):
           end = content.index("---", 3)
           return content[end + 3:].strip()
       return content
   ```

2. `read_resource(name, filename)` — Read resource file (L3 loading):
   ```python
   @tool(args_schema=ReadResourceInput)
   async def read_resource(name: str, filename: str) -> str:
       """Read a specific resource file from a skill directory."""
       # Validate filename doesn't escape directory (no ..)
       from pathlib import Path
       skills_dir = Path(__file__).parent.parent / "skills"
       resource = (skills_dir / name / filename).resolve()
       if not str(resource).startswith(str(skills_dir.resolve())):
           return json.dumps({"error": "Invalid path"})
       if not resource.exists():
           return json.dumps({"error": f"Resource '{filename}' not found in skill '{name}'"})
       return resource.read_text(encoding="utf-8")
   ```
