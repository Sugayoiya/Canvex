   ```python
   import contextvars
   import copy
   from dataclasses import dataclass

   @dataclass
   class ToolContext:
       project_id: str
       user_id: str
       team_id: str | None = None
       canvas_id: str | None = None
       episode_id: str | None = None

   _tool_ctx_var: contextvars.ContextVar[ToolContext | None] = contextvars.ContextVar("_tool_ctx_var", default=None)

   def set_tool_context(project_id: str, user_id: str, team_id: str | None = None, canvas_id: str | None = None, episode_id: str | None = None) -> contextvars.Token:
       """Set tool context. Returns token for reset. MUST be called inside the request's async context."""
       return _tool_ctx_var.set(ToolContext(project_id=project_id, user_id=user_id, team_id=team_id, canvas_id=canvas_id, episode_id=episode_id))

   def get_tool_context() -> ToolContext:
       ctx = _tool_ctx_var.get()
       if ctx is None:
           raise RuntimeError("Tool context not set — call set_tool_context() before agent run")
       return ctx

   def clear_tool_context() -> None:
       _tool_ctx_var.set(None)

   def get_context_snapshot() -> contextvars.Context:
       """Snapshot current context for safe propagation across async boundaries (e.g. asyncio.create_task).
       Usage: ctx = get_context_snapshot(); ctx.run(some_function)"""
       return contextvars.copy_context()
   ```

   ```python
   """Tool gating middleware: context-aware tool subset + standard timeout wrapper.

   Addresses review concern: 17 tools near LLM reliability cliff.
   Solution: expose ≤14 tools per agent invocation based on session context.
   """
   import asyncio
   import functools
   import logging
   from typing import Callable

   logger = logging.getLogger(__name__)

   TOOL_TIMEOUT_DEFAULT = 60   # seconds for read/write/skill tools
   TOOL_TIMEOUT_AI = 120       # seconds for generate_image/generate_video

   _AI_TOOL_NAMES = {"generate_image", "generate_video"}

   # Tools always included (core read + skill meta)
   _ALWAYS_TOOLS = {
       "get_project_info", "get_episodes", "get_script",
       "get_characters", "get_scenes",
       "read_skill", "read_resource",
   }

   # Tools included only when canvas_id is present
   _CANVAS_TOOLS = {"get_canvas_state"}

   # Tools included only when episode context is present (storyboard work)
   _STORYBOARD_TOOLS = {
       "save_shot_plan", "save_shot_details", "update_shot",
       "generate_image", "generate_video",
       "get_style_templates",
   }

   # Tools included for script/character/scene work
   _SCRIPT_TOOLS = {
       "save_characters", "save_scenes", "save_screenplay",
   }

   def get_tools_for_context(
       all_tools: list,
       *,
       has_canvas: bool = False,
       has_episode: bool = False,
   ) -> list:
       """Return ≤14 tools based on session context.

       Default set: 7 core + script tools (10).
       With canvas: +1 (11).
       With episode/storyboard: +6 (up to 14 max with overlap).
       """
       allowed_names = set(_ALWAYS_TOOLS) | set(_SCRIPT_TOOLS)

       if has_canvas:
           allowed_names |= _CANVAS_TOOLS
       if has_episode:
           allowed_names |= _STORYBOARD_TOOLS

       selected = [t for t in all_tools if t.name in allowed_names]

       if len(selected) > 14:
           logger.warning("Tool count %d exceeds target 14; consider refining gating rules", len(selected))

       return selected

   def tool_timeout(timeout_seconds: int | None = None):
       """Decorator: wrap an async tool function with asyncio.wait_for timeout.

       If timeout_seconds is None, uses TOOL_TIMEOUT_AI for AI tools, TOOL_TIMEOUT_DEFAULT otherwise.
       """
       def decorator(fn: Callable):
           @functools.wraps(fn)
           async def wrapper(*args, **kwargs):
               t = timeout_seconds
               if t is None:
                   t = TOOL_TIMEOUT_AI if fn.__name__ in _AI_TOOL_NAMES else TOOL_TIMEOUT_DEFAULT
               try:
                   return await asyncio.wait_for(fn(*args, **kwargs), timeout=t)
               except asyncio.TimeoutError:
                   import json
                   return json.dumps({"error": f"Tool '{fn.__name__}' timed out after {t}s"}, ensure_ascii=False)
           return wrapper
       return decorator
   ```

   ```python
   @tool(args_schema=ReadSkillInput)
   async def read_skill(name: str) -> str:
       """Read the detailed instructions for a specific skill. Call this before executing a complex workflow."""
       from pathlib import Path
       skills_dir = Path(__file__).parent.parent / "skills"
       skill_md = skills_dir / name / "SKILL.md"
       if not skill_md.exists():
           return json.dumps({"error": f"Skill '{name}' not found"})
       content = skill_md.read_text(encoding="utf-8")
       if content.startswith("---"):
           end = content.index("---", 3)
           return content[end + 3:].strip()
       return content
   ```
