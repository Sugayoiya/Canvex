```python
import logging
import yaml
from pathlib import Path
from typing import NamedTuple

logger = logging.getLogger(__name__)

class SkillMeta(NamedTuple):
    name: str
    description: str
    path: Path

class SkillLoader:
    """Three-level progressive disclosure for Anthropic SKILL.md directories."""

    def __init__(self, skills_dir: Path | None = None):
        self.skills_dir = skills_dir or (Path(__file__).parent / "skills")
        self._metadata: dict[str, SkillMeta] = {}

    def load_metadata(self) -> list[SkillMeta]:
        """L1: Parse YAML frontmatter from all SKILL.md files at startup."""
        self._metadata.clear()
        if not self.skills_dir.exists():
            logger.warning("Skills directory not found: %s", self.skills_dir)
            return []
        for skill_dir in sorted(self.skills_dir.iterdir()):
            if not skill_dir.is_dir():
                continue
            skill_md = skill_dir / "SKILL.md"
            if not skill_md.exists():
                continue
            try:
                meta = self._parse_frontmatter(skill_md)
                self._metadata[meta.name] = meta
                logger.debug("Loaded skill metadata: %s", meta.name)
            except Exception:
                logger.exception("Failed to parse SKILL.md: %s", skill_md)
        logger.info("SkillLoader: loaded %d skill(s)", len(self._metadata))
        return list(self._metadata.values())

    def build_system_prompt_fragment(self) -> str:
        """Build 'Available Skills' section for system prompt from L1 cache."""
        if not self._metadata:
            return ""
        lines = ["## 可用技能\n", "使用 `read_skill(name)` 获取技能详细指令后再执行。\n"]
        for m in self._metadata.values():
            lines.append(f"- **{m.name}**: {m.description}")
        return "\n".join(lines)

    def read_skill(self, name: str) -> str:
        """L2: Return full SKILL.md body (Markdown after frontmatter)."""
        skill_md = self.skills_dir / name / "SKILL.md"
        if not skill_md.exists():
            return f"Skill '{name}' not found."
        content = skill_md.read_text(encoding="utf-8")
        if content.startswith("---"):
            try:
                end = content.index("---", 3)
                return content[end + 3:].strip()
            except ValueError:
                return content
        return content

    def read_resource(self, name: str, filename: str) -> str:
        """L3: Return content of a specific resource file in skill directory."""
        resource = (self.skills_dir / name / filename).resolve()
        if not str(resource).startswith(str(self.skills_dir.resolve())):
            return "Error: invalid path"
        if not resource.exists():
            return f"Resource '{filename}' not found in skill '{name}'."
        return resource.read_text(encoding="utf-8")

    def list_skills(self) -> list[dict]:
        """Return list of {name, description} for API listing."""
        return [{"name": m.name, "description": m.description} for m in self._metadata.values()]

    def _parse_frontmatter(self, path: Path) -> SkillMeta:
        """Parse YAML frontmatter between --- delimiters."""
        text = path.read_text(encoding="utf-8")
        if not text.startswith("---"):
            raise ValueError(f"No YAML frontmatter in {path}")
        end = text.index("---", 3)
        fm = yaml.safe_load(text[3:end])
        return SkillMeta(
            name=fm["name"],
            description=fm.get("description", "").strip(),
            path=path.parent,
        )
```

**10 SKILL.md files:**

1. **extract-characters**: name=`extract-characters`. Workflow: `get_script(episode_id)` → Agent analyzes text → `save_characters(project_id, characters_json)`. Output format: JSON array `[{name, description, gender, age, personality}]`. Tips: from `_SYSTEM_PROMPT` in characters.py — focus on characters with dialogue, infer gender/age from context.

2. **extract-scenes**: name=`extract-scenes`. Workflow: `get_script(episode_id)` → Agent identifies scenes → `save_scenes(project_id, scenes_json)`. Output: `[{name, description, location, time_of_day, mood}]`.

3. **split-clips**: name=`split-clips`. Workflow: User provides story text → Agent splits into clip segments → return structured clips. Output: `[{clip_number, content, summary}]`.

4. **convert-screenplay**: name=`convert-screenplay`. Workflow: Takes clip content → Agent converts to screenplay format with scene headings, action lines, dialogue. Output: formatted screenplay text.

5. **create-storyboard**: name=`create-storyboard`. Workflow: `get_script(episode_id)` + optionally `get_characters` + `get_scenes` → Agent plans shots → `save_shot_plan(episode_id, shots_json)`. Output: `[{shot_number, description, characters, scene, dialogue}]`. Tips: from plan.py — ensure shot_number sequence is continuous.

6. **detail-storyboard**: name=`detail-storyboard`. Workflow: reads existing shots → Agent enriches each with camera angle, lighting, composition details → `save_shot_details(shot_id, details_json)`. Output: enriched shot data.

7. **generate-shot-image**: name=`generate-shot-image`. Workflow: Agent reads shot details → composes Imagen prompt → `generate_image(prompt, aspect_ratio)` → `update_shot(shot_id, "image_url", url)`. Tips: from character_prompt.py — include appearance, expression, pose, background.

8. **generate-shot-video**: name=`generate-shot-video`. Workflow: Agent reads shot + image → composes video prompt → `generate_video(prompt, aspect_ratio, image_url)`. Output: video url.

9. **episode-pipeline**: name=`episode-pipeline`. Workflow: full 4-step pipeline: `read_skill("split-clips")` → execute → `read_skill("convert-screenplay")` → execute → `read_skill("create-storyboard")` → execute → `read_skill("detail-storyboard")` → execute. This is the meta-skill that orchestrates the full episode creation flow. Tips: from pipeline_tools.py step order.

10. **refine-text**: name=`refine-text`. Workflow: User provides text + style direction → Agent refines directly (no tool needed — Agent IS the LLM) → return polished text. This is the simplest skill — Agent just does the refinement itself.

Each SKILL.md body should be 30-80 lines. Do NOT include Python code — only natural language instructions that the Agent follows.
