Create store with `create` from `zustand` (no persist — chat state is session-scoped, not persisted to localStorage):

```typescript
export const useChatStore = create<ChatState>()((set, get) => ({
  isOpen: false,
  sessionId: null,
  projectId: null,
  canvasId: null,
  messages: [],
  isStreaming: false,
  thinkingText: null,

  toggle: () => set((s) => ({ isOpen: !s.isOpen })),
  open: (projectId, canvasId) => set({ isOpen: true, projectId, canvasId: canvasId ?? null }),
  close: () => set({ isOpen: false }),
  setSession: (sessionId) => set({ sessionId, messages: [] }),
  addMessage: (message) => set((s) => ({ messages: [...s.messages, message] })),
  updateLastMessage: (content) => set((s) => {
    const msgs = [...s.messages];
    const last = msgs[msgs.length - 1];
    if (last && last.role === "assistant") {
      msgs[msgs.length - 1] = { ...last, content };
    }
    return { messages: msgs };
  }),
  setStreaming: (streaming) => set({ isStreaming: streaming }),
  setThinkingText: (text) => set({ thinkingText: text }),
  reset: () => set({ sessionId: null, messages: [], isStreaming: false, thinkingText: null }),
}));
```

Create `web/src/hooks/use-agent-chat.ts` — SSE streaming hook per RESEARCH.md Example 4 and UI-SPEC:

```typescript
import { useCallback, useRef } from "react";
import { fetchEventSource } from "@microsoft/fetch-event-source";
import { useChatStore, AgentMessage, ToolCallData, ToolResultData } from "@/stores/chat-store";
import { API_BASE_URL } from "@/lib/api";
```

**Hook: `useAgentChat()`**

Returns: `{ sendMessage, abort, isStreaming }`

Implementation:

```typescript
export function useAgentChat() {
  const abortRef = useRef<AbortController | null>(null);
  const store = useChatStore();
  // Read isStreaming from store
  const isStreaming = useChatStore((s) => s.isStreaming);

  const sendMessage = useCallback(async (content: string) => {
    const { sessionId, addMessage, updateLastMessage, setStreaming, setThinkingText } = useChatStore.getState();
    if (!sessionId || !content.trim()) return;

    const controller = new AbortController();
    abortRef.current = controller;
    setStreaming(true);
    setThinkingText(null);

    // Add user message optimistically
    const userMsg: AgentMessage = {
      id: `user-${Date.now()}`,
      role: "user",
      content: content.trim(),
      timestamp: Date.now(),
    };
    addMessage(userMsg);

    let assistantContent = "";
    let hasAssistantMessage = false;

    try {
      await fetchEventSource(`${API_BASE_URL}/api/v1/agent/chat/${sessionId}`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${localStorage.getItem("access_token")}`,
        },
        body: JSON.stringify({ message: content.trim() }),
        signal: controller.signal,

        onmessage(ev) {
          if (!ev.data) return;
          const data = JSON.parse(ev.data);

          switch (ev.event) {
            case "thinking":
              setThinkingText(data.status === "analyzing" ? "正在分析你的需求…" : "正在规划接下来的步骤…");
              break;

            case "tool_call": {
              setThinkingText(null);
              const toolCall: ToolCallData = { tool: data.tool, args: data.args, callId: data.call_id };
              const tcMsg: AgentMessage = {
                id: `tc-${data.call_id}`,
                role: "tool-call",
                content: `调用 ${data.tool}`,
                toolCalls: [toolCall],
                timestamp: Date.now(),
              };
              addMessage(tcMsg);
              break;
            }

            case "tool_result": {
              const toolResult: ToolResultData = { tool: data.tool, summary: data.summary, callId: data.call_id, success: data.success };
              const trMsg: AgentMessage = {
                id: `tr-${data.call_id}`,
                role: "tool-result",
                content: data.summary,
                toolResults: [toolResult],
                timestamp: Date.now(),
              };
              addMessage(trMsg);
              break;
            }

            case "token":
              setThinkingText(null);
              assistantContent += data.text;
              if (!hasAssistantMessage) {
                hasAssistantMessage = true;
                addMessage({
                  id: `asst-${Date.now()}`,
                  role: "assistant",
                  content: assistantContent,
                  timestamp: Date.now(),
                });
              } else {
                updateLastMessage(assistantContent);
              }
              break;

            case "done":
              setThinkingText(null);
              // Final content is already streamed via token events
              break;

            case "error":
              setThinkingText(null);
              addMessage({
                id: `err-${Date.now()}`,
                role: "assistant",
                content: `❌ ${data.message}`,
                timestamp: Date.now(),
              });
              break;

            case "heartbeat":
              // Keep-alive, no action
              break;
          }
        },

        onclose() {
          setStreaming(false);
          setThinkingText(null);
        },

        onerror(err) {
          setStreaming(false);
          setThinkingText(null);
          // Don't throw — let the stream close gracefully
          // fetchEventSource will auto-retry unless we throw
          throw err; // stop retry by throwing
        },

        openWhenHidden: true, // keep connection when tab is backgrounded
      });
    } catch (err) {
      // AbortError is expected when user cancels
      if (err instanceof DOMException && err.name === "AbortError") {
        // User cancelled — normal flow
      } else {
        console.error("Agent chat error:", err);
      }
    } finally {
      setStreaming(false);
      setThinkingText(null);
      abortRef.current = null;
    }
  }, []);

  const abort = useCallback(() => {
    abortRef.current?.abort();
    useChatStore.getState().setStreaming(false);
    useChatStore.getState().setThinkingText(null);
  }, []);

  return { sendMessage, abort, isStreaming };
}
```

Key implementation details:
- Use `useChatStore.getState()` inside callback to avoid stale closures (Zustand external access pattern)
- `openWhenHidden: true` prevents connection drop when user switches tabs
- AbortError handling distinguishes user-initiated cancel from real errors
- Token events accumulate into `assistantContent` string; first token creates message, subsequent tokens update it
- `throw err` in `onerror` stops fetchEventSource's auto-retry (we don't want infinite retries)
