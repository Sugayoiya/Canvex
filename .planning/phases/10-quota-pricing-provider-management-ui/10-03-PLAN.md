**Smart price display function** (per D-10):
```typescript
function smartPriceDisplay(rule: PricingRule): React.ReactNode {
  switch (rule.pricing_model) {
    case "per_token":
      return `In: $${rule.input_price_per_1k ?? 0} / Out: $${rule.output_price_per_1k ?? 0}`;
    case "fixed_request":
      return `Fix: $${rule.price_per_request ?? 0}`;
    case "per_image":
      return `Fix: $${rule.price_per_image ?? 0} Per Image`;
    case "per_second":
      return `$${rule.price_per_second ?? 0}/sec`;
    default:
      return "—";
  }
}
```
