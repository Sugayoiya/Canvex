export const aiProvidersApi = {
  list: (params?: { owner_type?: string; owner_id?: string }) => api.get("/ai-providers/", { params }),
  get: (providerId: string) => api.get(`/ai-providers/${providerId}`),
  create: (data: { provider_name: string; display_name: string; owner_type: string; owner_id?: string }) => api.post("/ai-providers/", data),
  update: (providerId: string, data: Record<string, unknown>) => api.patch(`/ai-providers/${providerId}`, data),
  delete: (providerId: string) => api.delete(`/ai-providers/${providerId}`),
  addKey: (providerId: string, data: { api_key: string; label?: string }) => api.post(`/ai-providers/${providerId}/keys`, data),
  deleteKey: (providerId: string, keyId: string) => api.delete(`/ai-providers/${providerId}/keys/${keyId}`),
};
```

const createMutation = useMutation({
  mutationFn: (data: Record<string, unknown>) => aiProvidersApi.create(data as any),
  onSuccess: (_, variables) => {
    queryClient.invalidateQueries({ queryKey: ["admin", "providers"] });
    toast.success(`已添加 Provider: ${(variables as any).display_name}`);
    setFormModal({ isOpen: false, editData: null });
  },
  onError: (err: unknown) => {
    const msg = (err as any)?.response?.data?.detail ?? "Unknown error";
    toast.error(`添加 Provider 失败: ${msg}`);
  },
});
```
