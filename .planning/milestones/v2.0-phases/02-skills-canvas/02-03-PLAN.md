    0. **FIRST** — Create shared JSON parser utility `api/app/skills/utils/json_parser.py` (addresses review HIGH: robust JSON parsing across all skills):
       ```python
       import json
       import re
       import logging
       from typing import Any

       logger = logging.getLogger(__name__)

       class LLMJsonParseError(Exception):
           """Raised when LLM output cannot be parsed as valid JSON."""
           def __init__(self, message: str, raw_output: str = ""):
               super().__init__(message)
               self.raw_output = raw_output

       def parse_llm_json(raw: str, *, wrapper_key: str | None = None) -> Any:
           """Robust JSON parser for LLM outputs.

           Pipeline: strip fences → find JSON substring → parse → unwrap key if needed.
           Raises LLMJsonParseError with raw_output for diagnostics.
           """
           # Step 1: Strip markdown fences (```json ... ```)
           match = re.search(r'```(?:json)?\s*\n?(.*?)\n?\s*```', raw, re.DOTALL)
           text = match.group(1).strip() if match else raw.strip()

           # Step 2: Try direct parse
           try:
               parsed = json.loads(text)
           except json.JSONDecodeError:
               # Step 3: Recover JSON substring — find first [ or { and last ] or }
               json_match = re.search(r'[\[{]', text)
               if json_match:
                   start = json_match.start()
                   # Find matching close
                   for end in range(len(text), start, -1):
                       try:
                           parsed = json.loads(text[start:end])
                           break
                       except json.JSONDecodeError:
                           continue
                   else:
                       raise LLMJsonParseError(f"No valid JSON found in LLM output ({len(raw)} chars)", raw_output=raw[:500])
               else:
                   raise LLMJsonParseError(f"No JSON delimiters found in LLM output", raw_output=raw[:500])

           # Step 4: Unwrap wrapper key if present
           if wrapper_key and isinstance(parsed, dict) and wrapper_key in parsed:
               parsed = parsed[wrapper_key]

           return parsed
       ```

                 data={"characters": characters, "warnings": warnings},
