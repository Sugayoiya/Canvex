    **1c. api/app/services/quota_service.py — Fail-closed QuotaService with atomic check**

    Core methods:
    - `check_quota(user_id, team_id)` — Pre-execution check. Returns `QuotaCheckResult(allowed=False)` on ANY error (fail-closed per user decision). Uses `SELECT ... FOR UPDATE` to lock the row during check.
    - `update_usage(user_id, team_id, skill_execution_id, credit_amount)` — Post-execution increment. Idempotent by `skill_execution_id` unique constraint on `QuotaUsageLog`. Uses `SELECT ... FOR UPDATE` for atomic counter increment.
    - `_lazy_reset(quota_row, now)` — Resets counters if month/day has changed since `last_month_reset`/`last_day_reset`.

    Critical implementation rules:
    - **Fail-closed**: `check_quota` wraps entire logic in try/except. On ANY exception (DB error, timeout), return `QuotaCheckResult(allowed=False, reason="配额检查服务暂时不可用", error_code="QUOTA_SERVICE_ERROR")`.
    - **Atomic increment**: `update_usage` uses `with_for_update()` on the SELECT query to prevent concurrent double-count.
    - **Idempotent**: `update_usage` first checks if a `QuotaUsageLog` with the given `skill_execution_id` already exists. If yes, skip the increment (Celery retry safety).
    - **Policy order**: Check user quota first, then team quota. First failure wins.
    - **NULL limit = unlimited**: If `monthly_credit_limit` is NULL, that dimension is not enforced.

    ```python
    import logging
    from datetime import datetime
    from decimal import Decimal
    from sqlalchemy import select
    from app.core.database import AsyncSessionLocal
    from app.models.quota import UserQuota, TeamQuota, QuotaUsageLog
    from app.schemas.quota import QuotaCheckResult

    logger = logging.getLogger(__name__)

    class QuotaService:
        @staticmethod
        async def check_quota(user_id: str, team_id: str | None = None) -> QuotaCheckResult:
            try:
                async with AsyncSessionLocal() as session:
                    # Check user quota with row lock
                    stmt = select(UserQuota).where(UserQuota.user_id == user_id).with_for_update()
                    uq = (await session.execute(stmt)).scalar_one_or_none()
                    if uq:
                        QuotaService._lazy_reset(uq, datetime.utcnow())
                        if uq.monthly_credit_limit is not None and uq.current_month_usage >= uq.monthly_credit_limit:
                            return QuotaCheckResult(allowed=False, reason="用户月度额度已用完", error_code="USER_MONTHLY_QUOTA_EXCEEDED", current_usage=uq.current_month_usage, limit=uq.monthly_credit_limit)
                        if uq.daily_call_limit is not None and uq.current_day_calls >= uq.daily_call_limit:
                            return QuotaCheckResult(allowed=False, reason="用户每日调用次数已达上限", error_code="USER_DAILY_QUOTA_EXCEEDED")
                    # Check team quota
                    if team_id:
                        stmt = select(TeamQuota).where(TeamQuota.team_id == team_id).with_for_update()
                        tq = (await session.execute(stmt)).scalar_one_or_none()
                        if tq:
                            QuotaService._lazy_reset(tq, datetime.utcnow())
                            if tq.monthly_credit_limit is not None and tq.current_month_usage >= tq.monthly_credit_limit:
                                return QuotaCheckResult(allowed=False, reason="团队月度额度已用完", error_code="TEAM_MONTHLY_QUOTA_EXCEEDED", current_usage=tq.current_month_usage, limit=tq.monthly_credit_limit)
                            if tq.daily_call_limit is not None and tq.current_day_calls >= tq.daily_call_limit:
                                return QuotaCheckResult(allowed=False, reason="团队每日调用次数已达上限", error_code="TEAM_DAILY_QUOTA_EXCEEDED")
                    await session.commit()
                    return QuotaCheckResult(allowed=True)
            except Exception:
                logger.exception("Quota check failed — fail-closed")
                return QuotaCheckResult(allowed=False, reason="配额检查服务暂时不可用", error_code="QUOTA_SERVICE_ERROR")

        @staticmethod
        async def update_usage(user_id: str, team_id: str | None, skill_execution_id: str, credit_amount: Decimal) -> bool:
            """Post-execution usage increment. Idempotent by skill_execution_id. Returns True if incremented, False if duplicate."""
            try:
                async with AsyncSessionLocal() as session:
                    # (a) Idempotency guard — check if already recorded
                    existing = (await session.execute(
                        select(QuotaUsageLog).where(QuotaUsageLog.skill_execution_id == skill_execution_id)
                    )).scalar_one_or_none()
                    if existing:
                        return False  # Already counted — Celery retry safe

                    now = datetime.utcnow()

                    # (b) Atomic increment user quota with row lock
                    stmt = select(UserQuota).where(UserQuota.user_id == user_id).with_for_update()
                    uq = (await session.execute(stmt)).scalar_one_or_none()
                    if uq:
                        QuotaService._lazy_reset(uq, now)
                        uq.current_month_usage += credit_amount
                        uq.current_day_calls += 1
                        uq.updated_at = now

                    # (c) Atomic increment team quota with row lock
                    if team_id:
                        stmt = select(TeamQuota).where(TeamQuota.team_id == team_id).with_for_update()
                        tq = (await session.execute(stmt)).scalar_one_or_none()
                        if tq:
                            QuotaService._lazy_reset(tq, now)
                            tq.current_month_usage += credit_amount
                            tq.current_day_calls += 1
                            tq.updated_at = now

                    # (d) Create usage log entry (unique constraint on skill_execution_id prevents double-insert)
                    session.add(QuotaUsageLog(
                        user_id=user_id,
                        team_id=team_id,
                        skill_execution_id=skill_execution_id,
                        credit_amount=credit_amount,
                        action="increment",
                    ))

                    # (e) Commit atomically
                    await session.commit()
                    return True
            except Exception:
                logger.exception("Quota update_usage failed for execution %s", skill_execution_id)
                return False

        @staticmethod
        def _lazy_reset(quota, now: datetime):
            if quota.last_month_reset.month != now.month or quota.last_month_reset.year != now.year:
                quota.current_month_usage = Decimal("0")
                quota.last_month_reset = now
            if quota.last_day_reset.date() != now.date():
                quota.current_day_calls = 0
                quota.last_day_reset = now
    ```
