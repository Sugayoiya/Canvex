Create `web/src/components/canvas/hooks/use-upstream-data.ts` with the following implementation:

```typescript
"use client";

import { useMemo } from "react";
import { useEdges, useNodes } from "@xyflow/react";

export interface UpstreamData {
  text: string[];
  json: object[];
  imageUrl: string[];
}

const EMPTY: UpstreamData = { text: [], json: [], imageUrl: [] };

export function useUpstreamData(nodeId: string): UpstreamData {
  const edges = useEdges();
  const nodes = useNodes();

  return useMemo(() => {
    const incoming = edges.filter((e) => e.target === nodeId);
    if (incoming.length === 0) return EMPTY;

    // DETERMINISTIC ORDER (review concern #6): sort by source node ID
    // so multi-upstream aggregation produces consistent array ordering
    // regardless of edge creation order or internal ReactFlow shuffling.
    const sorted = [...incoming].sort((a, b) => a.source.localeCompare(b.source));

    const result: UpstreamData = { text: [], json: [], imageUrl: [] };

    for (const edge of sorted) {
      const src = nodes.find((n) => n.id === edge.source);
      if (!src) continue;
      const d = src.data as Record<string, unknown>;

      // Text: from text-input (data.text) or LLM result (data.result_text)
      if (typeof d.text === "string" && d.text) result.text.push(d.text);
      if (typeof d.result_text === "string" && d.result_text)
        result.text.push(d.result_text);

      // JSON: from extract results (data.result_data)
      if (d.result_data && typeof d.result_data === "object")
        result.json.push(d.result_data as object);

      // Image URLs: from image-gen results (data.result_url)
      if (typeof d.result_url === "string" && d.result_url)
        result.imageUrl.push(d.result_url);
    }

    return result;
  }, [nodeId, edges, nodes]);
}
```

Key design decisions per D-07/D-08/D-09 + review feedback:
- Returns `{ text: string[], json: object[], imageUrl: string[] }` — three typed channels
- Multi-upstream same-type inputs aggregate into arrays (D-08)
- **Stable edge ordering (review concern #6):** incoming edges sorted by `edge.source` (localeCompare) before aggregation. This ensures consistent array indices across renders and re-connections. The ordering is deterministic — same graph topology always produces same array order.
- Source node data field mapping: `text-input` → `data.text`; `llm-generate` → `data.result_text`; `extract` → `data.result_data`; `image-gen` → `data.result_url`
- Nodes consume via `upstream.text[0] ?? ""` for single-value access or `upstream.text.join("\n")` for aggregation

**1. Create `web/src/components/canvas/hooks/use-node-persistence.ts`:**

Following OpenLovart `isSavingRef`/`needsSaveRef` pattern (D-12), adapted for per-node incremental updates (D-11).

**Conflict strategy (review concern #2):** This hook uses last-write-wins with write serialization via `isSavingRef`. Only one `canvasApi.updateNode` call is in-flight at a time per node. If a second write arrives during an active save, `needsSaveRef` queues it and retries after the current save completes. This prevents concurrent writes from interleaving. When execution writeback (from `use-node-execution.ts`) needs to write results, it calls `cancelPending()` first to discard any stale debounced config save, then writes results directly — results always take priority over in-progress config edits.

**Tab close safety (review concern #4):** A `beforeunload` event listener calls `flush()` to persist any pending debounced data before the page unloads.

**Graceful degradation (review concern #3):** `doSave` catches errors and logs them without throwing. The UI continues working even if persistence fails — the data stays in React state and will be retried on next edit.

```typescript
"use client";

import { useCallback, useEffect, useRef } from "react";
import { canvasApi } from "@/lib/api";

const DEBOUNCE_MS = 2000;

export function useNodePersistence(nodeId: string) {
  const isSavingRef = useRef(false);
  const needsSaveRef = useRef(false);
  const pendingDataRef = useRef<Record<string, unknown> | null>(null);
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  const doSave = useCallback(
    async (data: Record<string, unknown>) => {
      if (isSavingRef.current) {
        needsSaveRef.current = true;
        pendingDataRef.current = data;
        return;
      }
      isSavingRef.current = true;
      try {
        await canvasApi.updateNode(nodeId, data);
      } catch (err) {
        console.warn(`[useNodePersistence] save failed for node ${nodeId}:`, err);
      } finally {
        isSavingRef.current = false;
        if (needsSaveRef.current && pendingDataRef.current) {
          needsSaveRef.current = false;
          const next = pendingDataRef.current;
          pendingDataRef.current = null;
          doSave(next);
        }
      }
    },
    [nodeId],
  );

  const saveDebounced = useCallback(
    (data: Record<string, unknown>) => {
      pendingDataRef.current = data;
      if (timerRef.current) clearTimeout(timerRef.current);
      timerRef.current = setTimeout(() => {
        if (pendingDataRef.current) {
          doSave(pendingDataRef.current);
        }
      }, DEBOUNCE_MS);
    },
    [doSave],
  );

  const saveImmediate = useCallback(
    (data: Record<string, unknown>) => {
      if (timerRef.current) {
        clearTimeout(timerRef.current);
        timerRef.current = null;
      }
      pendingDataRef.current = null;
      doSave(data);
    },
    [doSave],
  );

  const flush = useCallback(() => {
    if (timerRef.current) {
      clearTimeout(timerRef.current);
      timerRef.current = null;
    }
    if (pendingDataRef.current) {
      doSave(pendingDataRef.current);
      pendingDataRef.current = null;
    }
  }, [doSave]);

  const cancelPending = useCallback(() => {
    if (timerRef.current) {
      clearTimeout(timerRef.current);
      timerRef.current = null;
    }
    pendingDataRef.current = null;
    needsSaveRef.current = false;
  }, []);

  // beforeunload: flush pending writes when user closes/refreshes tab
  useEffect(() => {
    const handleBeforeUnload = () => {
      if (timerRef.current) {
        clearTimeout(timerRef.current);
        timerRef.current = null;
      }
      if (pendingDataRef.current) {
        // Use sendBeacon for reliable unload persistence
        const payload = JSON.stringify(pendingDataRef.current);
        const url = `/api/v1/canvas/nodes/${nodeId}`;
        const blob = new Blob([payload], { type: "application/json" });
        navigator.sendBeacon(url, blob);
        pendingDataRef.current = null;
      }
    };
    window.addEventListener("beforeunload", handleBeforeUnload);
    return () => {
      window.removeEventListener("beforeunload", handleBeforeUnload);
      if (timerRef.current) clearTimeout(timerRef.current);
    };
  }, [nodeId]);

  return {
    saveDebounced,
    saveImmediate,
    flush,
    cancelPending,
    isSaving: isSavingRef,
  };
}
```

Five methods per D-11 + review feedback:
- `saveDebounced(data)` — 2s debounce for text/config typing
- `saveImmediate(data)` — immediate for execute/connect/delete
- `flush()` — for blur events, flushes pending debounced save
- `cancelPending()` — for execution writeback: discard stale pending edits before writing results (review concern #2)
- `isSaving` — ref for UI indicators
